157  template<typename TFunc>
158  void effect(uint_fast16_t x, uint_fast16_t y, uint_fast16_t w, uint_fast16_t h, TFunc&& effector)
159  {
160  auto ye = y + h;
161  auto buf = (RGBColor*)alloca(w * sizeof(RGBColor));
162 #pragma GCC diagnostic push
163 #if defined(__has_warning)
164 #if __has_warning("-Wmaybe-uninitialized")
165 #pragma GCC diagnostic ignored "-Wmaybe-uninitialized"
166 #endif
167 #else
168 #pragma GCC diagnostic ignored "-Wmaybe-uninitialized"
169 #endif
170  pixelcopy_t pc_write(buf ,_write_depth, RGBColor::depth, false);
172 #pragma GCC diagnostic pop
173  pixelcopy_t pc_read( nullptr, RGBColor::depth, _read_depth, false);
174  startWrite();
175  do
176  {
177  readRect(x, y, w, 1, buf, &pc_read);
178  size_t i = 0;
179  do
180  {
181  effector(x + i, y, buf[i]);
182  } while (++i < w);
183  writeImage(x, y, w, 1, &pc_write, true);
184  pc_write.src_x32 = 0;
185  } while (++y < ye);
186  endWrite();
187  }
188  };
