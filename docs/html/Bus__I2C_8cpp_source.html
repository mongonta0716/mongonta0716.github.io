129  void Bus_I2C::dc_control(bool dc)
130  {
131  // リード中の場合はトランザクションを終了しておく;
132  if (_state == state_t::state_read)
133  {
134  _state = state_t::state_none;
135  lgfx::i2c::endTransaction(_cfg.i2c_port);
136  }
137 
138  // まだトランザクションが開始されていない場合は開始しておく;
139  if (_state == state_t::state_none)
140  {
141  _state = state_t::state_write_none;
142  lgfx::i2c::beginTransaction(_cfg.i2c_port, _cfg.i2c_addr, _cfg.freq_write, false);
143  }
144 
145  // DCプリフィクスなしの場合は後の処理は不要;
146  if (_cfg.prefix_len == 0) return;
147 
148  state_t st = dc ? state_t::state_write_data : state_t::state_write_cmd;
149  // 既に送信済みのDCプリフィクスが要求と一致している場合は終了;
150  if (_state == st) return;
151 
152  // DCプリフィクスが送信済みの場合、送信済みのDCプリフィクスと要求が不一致なのでトランザクションをやり直す。;
153  int retry = 3;
154  do
155  {
156  if (_state != state_t::state_write_none)
157  {
158  lgfx::i2c::endTransaction(_cfg.i2c_port);
159  lgfx::i2c::beginTransaction(_cfg.i2c_port, _cfg.i2c_addr, _cfg.freq_write, false);
160  }
161  _state = st;
162  } while (lgfx::i2c::writeBytes(_cfg.i2c_port, (uint8_t*)(dc ? &_cfg.prefix_data : &_cfg.prefix_cmd), _cfg.prefix_len).has_error() && --retry);
163  if (!retry) { _state = state_none; }
164  }

178  void Bus_I2C::writeDataRepeat(uint32_t data, uint_fast8_t bit_length, uint32_t length)
179  {
180  dc_control(true);
181  const uint8_t dst_bytes = bit_length >> 3;
182  uint32_t buf0 = data | data << bit_length;
183  uint32_t buf1;
184  uint32_t buf2;
185  // make 12Bytes data.
186  if (dst_bytes != 3)
187  {
188  if (dst_bytes == 1)
189  {
190  buf0 |= buf0 << 16;
191  }
192  buf1 = buf0;
193  buf2 = buf0;
194  }
195  else
196  {
197  buf1 = buf0 >> 8 | buf0 << 16;
198  buf2 = buf0 >> 16 | buf0 << 8;
199  }
200  uint32_t src[8] = { buf0, buf1, buf2, buf0, buf1, buf2, buf0, buf1 };
201  auto buf = reinterpret_cast<uint8_t*>(src);
202  uint32_t limit = 32 / dst_bytes;
203  uint32_t len;
204  do
205  {
206  len = ((length - 1) % limit) + 1;
207  i2c::writeBytes(_cfg.i2c_port, buf, len * dst_bytes);
208  } while (length -= len);
209  }
