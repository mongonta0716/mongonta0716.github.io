167 static int32_t create_qt_tbl ( /* 0:OK, !0:Failed */
168  lgfxJdec* jd, /* Pointer to the decompressor object */
169  const uint8_t* data, /* Pointer to the quantizer tables */
170  uint_fast16_t ndata /* Size of input data */
171 )
172 {
173  const uint8_t* dataend = data + ndata;
174  do { /* Process all tables in the segment */
175  size_t d = *data++; /* Get table property */
176  if (d & 0xF0) return JDR_FMT1; /* Err: not 8-bit resolution */
177  int32_t *pb = (int32_t*)alloc_pool(jd, 64 * sizeof (int32_t));/* Allocate a memory block for the table */
178  if (!pb) return JDR_MEM1; /* Err: not enough memory */
179  jd->qttbl[d & 3] = pb; /* Register the table */
180  for (size_t i = 0; i < 64; ++i) { /* Load the table */
181  uint_fast8_t z = Zig[i]; /* Zigzag-order to raster-order conversion */
182  pb[z] = (int32_t)((uint32_t)data[i] * Ipsf[z]); /* Apply scale factor of Arai algorithm to the de-quantizers */
183  }
184  } while (dataend != (data += 64));
185 
186  return JDR_OK;
187 }

196 static int32_t create_huffman_tbl ( /* 0:OK, !0:Failed */
197  lgfxJdec* jd, /* Pointer to the decompressor object */
198  const uint8_t* data, /* Pointer to the packed huffman tables */
199  int_fast16_t ndata /* Size of input data */
200 )
201 {
202  uint_fast16_t np;
203  uint8_t *pb, *pd;
204  uint16_t *ph;
205 
206  do { /* Process all tables in the segment */
207  uint_fast8_t d = *data++; /* Get table number and class */
208  if (d & 0xEE) return JDR_FMT1; /* Err: invalid class/number */
209  uint_fast8_t cls = d >> 4; /* class = dc(0)/ac(1), table number = 0/1 */
210  uint_fast8_t num = d & 0x0F;
211  pb = alloc_pool(jd, 16); /* Allocate a memory block for the bit distribution table */
212  if (!pb) return JDR_MEM1; /* Err: not enough memory */
213  jd->huffbits[num][cls] = pb - 1;
214  np = 0;
215  size_t i = 0;
216  do { /* Load number of patterns for 1 to 16-bit code */
217  np += (pb[i] = data[i]); /* Get sum of code words for each code */
218  } while (++i < 16);
219 
220  ph = (uint16_t*)alloc_pool(jd, np * sizeof (uint16_t));/* Allocate a memory block for the code word table */
221  if (!ph) return JDR_MEM1; /* Err: not enough memory */
222  jd->huffcode[num][cls] = ph - 1;
223  uint_fast16_t hc = 0;
224  i = 0;
225  do { /* Re-build huffman code word table */
226  size_t b = pb[i];
227  while (b--) *ph++ = hc++;
228  hc <<= 1;
229  } while (++i < 16);
230 
231  pd = alloc_pool(jd, np); /* Allocate a memory block for the decoded data */
232  if (!pd) return JDR_MEM1; /* Err: not enough memory */
233  jd->huffdata[num][cls] = pd - 1;
234 
235  memcpy(pd, data += 16, np); /* Load decoded data corresponds to each code ward */
236  data += np;
237  } while (ndata -= 17 + np);
238 
239  return JDR_OK;
240 }

250  lgfxJdec* jd, /* Pointer to the decompressor object */

264  if (dp == dpend) return 0 - (int32_t)JDR_INP; /* Err: read error or wrong stream termination */
265  jd->dpend = dpend;
266  }

275  }

278  }

280  msk += 8; /* Read from MSB */
281  } while (msk < nbit);

350 static void block_idct (
351  int32_t* src, /* Input block data (de-quantized and pre-scaled for Arai Algorithm) */
352  int16_t* dst /* Pointer to the destination to store the block as byte array */
353 )
354 {
355  const int32_t M13 = (int32_t)(1.41421*256), M2 = (int32_t)(1.08239*256), M4 = (int32_t)(2.61313*256), M5 = (int32_t)(1.84776*256);
356  int32_t v0, v1, v2, v3, v4, v5, v6, v7;
357  int32_t t10, t11, t12, t13;
358 
359  /* Process columns */
360  for (int i = 0; i < 8; ++i) {
361  /* Get and Process the odd elements */
362  v4 = src[8 * 7];
363  v5 = src[8 * 1];
364  v6 = src[8 * 5];
365  v7 = src[8 * 3];
366 
367  t10 = v5 - v4;
368  t11 = v5 + v4;
369  t12 = v6 - v7;
370  v7 += v6;
371  v5 = (t11 - v7) * M13 >> 8;
372  t13 = (t10 + t12) * M5 >> 8;
373  v6 = t13 - ((t12 * M4 >> 8) + (v7 += t11));
374  v4 = t13 - ((t10 * M2 >> 8) + (v5 -= v6));
375 
376  /* Get and Process the even elements */
377  v0 = src[8 * 0];
378  v2 = src[8 * 4];
379  t10 = v0 + v2;
380  t12 = v0 - v2;
381 
382  v1 = src[8 * 2];
383  v3 = src[8 * 6];
384  t11 = (v1 - v3) * M13 >> 8;
385  v3 += v1;
386  t11 -= v3;
387 
388  v0 = t10 + v3;
389  v3 = t10 - v3;
390  v1 = t12 + t11;
391  v2 = t12 - t11;
392 
393  /* Write-back transformed values */
394  src[8 * 0] = v0 + v7;
395  src[8 * 7] = v0 - v7;
396  src[8 * 1] = v1 + v6;
397  src[8 * 6] = v1 - v6;
398  src[8 * 2] = v2 + v5;
399  src[8 * 5] = v2 - v5;
400  src[8 * 3] = v3 + v4;
401  src[8 * 4] = v3 - v4;
402 
403  ++src; /* Next column */
404  }
405 
406  /* Process rows */
407  src -= 8;
408  for (int i = 0; i < 8; ++i) {
409  /* Get and Process the odd elements */
410  v4 = src[1];
411  v5 = src[7] + v4;
412  v4 = (v4 << 1) - v5;
413 
414  v6 = src[5];
415  v7 = src[3] + v6;
416  v6 = (v6 << 1) - v7;
417 
418  v7 += v5;
419  v5 = (v5 << 1) - v7;
420 
421  t13 = v4 + v6;
422  t13 = t13 * M5 >> 8;
423  v6 = v6 * M4 >> 8;
424  v6 += v7;
425  v6 = t13 - v6;
426  v5 = v5 * M13 >> 8;
427  v5 -= v6;
428  v4 = v4 * M2 >> 8;
429  v4 += v5;
430  v4 = t13 - v4;
431 
432  /* Get and Process the even elements */
433  v0 = src[0] + (128L << 8); /* remove DC offset (-128) here */
434  v2 = src[4];
435  t10 = v0 + v2;
436  t12 = v0 - v2;
437 
438  v1 = src[2];
439  v3 = src[6] + v1;
440  t11 = (v1 << 1) - v3;
441  t11 = t11 * M13 >> 8;
442  t11 -= v3;
443 
444  v0 = t10 + v3;
445  v3 = t10 - v3;
446  v1 = t12 + t11;
447  v2 = t12 - t11;
448 
449  dst[0] = (v0 + v7) >> 8;
450  dst[7] = (v0 - v7) >> 8;
451  dst[1] = (v1 + v6) >> 8;
452  dst[6] = (v1 - v6) >> 8;
453  dst[2] = (v2 + v5) >> 8;
454  dst[5] = (v2 - v5) >> 8;
455  dst[3] = (v3 + v4) >> 8;
456  dst[4] = (v3 - v4) >> 8;
457 
458  dst += 8;
459  src += 8; /* Next row */
460  }

470 static JRESULT mcu_load (
471  lgfxJdec* jd /* Pointer to the decompressor object */
472 )
473 {
474  int32_t *tmp = (int32_t*)jd->workbuf; /* Block working buffer for de-quantize and IDCT */
475  int32_t b, d, e;
476  uint32_t blk, nby, nbc;
477  int16_t *bp;
478  const uint8_t *hb, *hd;
479  const uint16_t *hc;
480 
481 
482  nby = jd->msx * jd->msy; /* Number of Y blocks (1, 2 or 4) */
483  nbc = jd->comps_in_frame - 1; /* Number of C blocks (2 or 0(grayscale)) */
484  bp = jd->mcubuf; /* Pointer to the first block */
485 
486  for (blk = 0; blk < nby + nbc; ++blk) {
487  size_t cmp = (blk < nby) ? 0 : blk - nby + 1; /* Component number 0:Y, 1:Cb, 2:Cr */
488  size_t id = cmp ? 1 : 0; /* Huffman table ID of the component */
489 
490  /* Extract a DC element from input stream */
491  hb = jd->huffbits[id][0]; /* Huffman table for the DC element */
492  hc = jd->huffcode[id][0];
493  hd = jd->huffdata[id][0];
494  b = huffext(jd, hb, hc, hd); /* Extract a huffman coded data (bit length) */
495  if (b < 0) return (JRESULT)(-b); /* Err: invalid code or input */
496  d = jd->dcv[cmp]; /* DC value of previous block */
497  if (b) { /* If there is any difference from previous block */
498  e = bitext(jd, b); /* Extract data bits */
499  if (e < 0) return (JRESULT)(-e); /* Err: input */
500  b = 1 << (b - 1); /* MSB position */
501  if (!(e & b)) e -= (b << 1) - 1; /* Restore sign if needed */
502  d += e; /* Get current value */
503  jd->dcv[cmp] = d; /* Save current DC value for next block */
504  }
505  const int32_t *dqf = jd->qttbl[jd->qtid[cmp]]; /* De-quantizer table ID for this component */
506  tmp[0] = d * dqf[0] >> 8; /* De-quantize, apply scale factor of Arai algorithm and descale 8 bits */
507 
508  /* Extract following 63 AC elements from input stream */
509  memset(&tmp[1], 0, 63*sizeof(int32_t)); /* Clear rest of elements */
510  hb = jd->huffbits[id][1]; /* Huffman table for the AC elements */
511  hc = jd->huffcode[id][1];
512  hd = jd->huffdata[id][1];
513  uint_fast8_t i = 1; /* Top of the AC elements */
514  do {
515  b = huffext(jd, hb, hc, hd); /* Extract a huffman coded value (zero runs and bit length) */
516  if (b == 0) break; /* EOB? */
517  if (b < 0) return (JRESULT)(-b); /* Err: invalid code or input error */
518  i += b >> 4; /* Number of leading zero elements Skip zero elements */
519  if (b &= 0x0F) { /* Bit length */
520  d = bitext(jd, b); /* Extract data bits */
521  if (d < 0) return (JRESULT)(-d);/* Err: input device */
522  b = 1 << (b - 1); /* MSB position */
523  if (!(d & b)) d -= (b << 1) - 1;/* Restore negative value if needed */
524  uint_fast8_t z = Zig[i]; /* Zigzag-order to raster-order converted index */
525  tmp[z] = d * dqf[z] >> 8; /* De-quantize, apply scale factor of Arai algorithm and descale 8 bits */
526  }
527  } while (++i < 64); /* Next AC element */
528 
529  if (i == 1 || (JD_USE_SCALE && jd->scale == 3)) {
530  d = (int16_t)((*tmp >> 8) + 128); /* If scale ratio is 1/8, IDCT can be ommited and only DC element is used */
531  for (i = 0; i < 64; bp[i++] = d) ;
532  } else {
533  block_idct(tmp, bp); /* Apply IDCT and store the block to the MCU buffer */
534  }
535 
536  bp += 64; /* Next block */
537  }
538 
539  return JDR_OK; /* All blocks have been loaded successfully */
540 }

545 /*-----------------------------------------------------------------------*/
546 /* Output an MCU: Convert YCrCb to RGB and output it in RGB form */
547 /*-----------------------------------------------------------------------*/
548 
549 static JRESULT mcu_output (
550  lgfxJdec* jd, /* Pointer to the decompressor object */
551  uint32_t (*outfunc)(void*, void*, JRECT*), /* RGB output function */
552  uint32_t x, /* MCU position in the image (left of the MCU) */
553  uint32_t y /* MCU position in the image (top of the MCU) */
554 )
555 {
556  const int_fast16_t FP_SHIFT = 8;
557  uint32_t ix, iy, mx, my, rx, ry;
558  int32_t yy, cb, cr;
559  int16_t *py, *pc;
560  uint8_t *rgb24;
561  JRECT rect;
562 
563  mx = jd->msx << 3; my = jd->msy << 3; /* MCU size (pixel) */
564  rx = (mx < jd->width - x) ? mx : jd->width - x; /* Output rectangular size (it may be clipped at right/bottom end) */
565  ry = (my < jd->height - y) ? my : jd->height - y;
566 
567  if (JD_USE_SCALE) {
568  rx >>= jd->scale; ry >>= jd->scale;
569  if (!rx || !ry) return JDR_OK; /* Skip this MCU if all pixel is to be rounded off */
570  x >>= jd->scale; y >>= jd->scale;
571  }
572  rect.left = x; rect.right = x + rx - 1; /* Rectangular area in the frame buffer */
573  rect.top = y; rect.bottom = y + ry - 1;
574 
575  uint8_t* workbuf = (uint8_t*)jd->workbuf;
576 
577  if (!JD_USE_SCALE || jd->scale != 3) { /* Not for 1/8 scaling */
578 
579  uint_fast8_t ixshift = (mx == 16);
580  uint_fast8_t iyshift = (my == 16);
581 
582  /* Build an RGB MCU from discrete comopnents */
583  rgb24 = workbuf;
584  iy = 0;
585  do {
586 #if JD_BAYER
587  const int_fast8_t* btbl = &Bayer[(iy & 3) << 2];
588 #endif
589  py = &jd->mcubuf[((iy & 8) + iy) << 3];
590  pc = &jd->mcubuf[((mx << iyshift) + (iy >> iyshift)) << 3];
591  ix = 0;
592  do {
593  do {
594  cb = (pc[ 0] - 128); /* Get Cb/Cr component and restore right level */
595  cr = (pc[64] - 128);
596  ++pc;
597 
598  /* Convert CbCr to RGB */
599  uint_fast16_t rr = ((int32_t)(1.402 * (1<<FP_SHIFT)) * cr) >> FP_SHIFT;
600  uint_fast16_t gg = ((int32_t)(0.34414 * (1<<FP_SHIFT)) * cb
601  + (int32_t)(0.71414 * (1<<FP_SHIFT)) * cr) >> FP_SHIFT;
602  uint_fast16_t bb = ((int32_t)(1.772 * (1<<FP_SHIFT)) * cb) >> FP_SHIFT;
603  do {
604 #if JD_BAYER
605  yy = *py + btbl[ix & 3]; /* Get Y component */
606 #else
607  yy = *py; /* Get Y component */
608 #endif
609  ++py;
610  /* Convert YCbCr to RGB */
611  rgb24[0] = BYTECLIP(yy + rr);
612  rgb24[1] = BYTECLIP(yy - gg);
613  rgb24[2] = BYTECLIP(yy + bb);
614  rgb24 += 3;
615  } while (++ix & ixshift);
616  } while (ix & 7);
617  py += 64 - 8; /* Jump to next block if double block heigt */
618  } while (ix != mx);
619  } while (++iy < my);
620 
621  /* Descale the MCU rectangular if needed */
622  if (JD_USE_SCALE && jd->scale) {
623  uint32_t x_, y_, r_, g_, b_, s_, w_;
624  uint8_t *op;
625 
626  /* Get averaged RGB value of each square correcponds to a pixel */
627  s_ = jd->scale * 2; /* Bumber of shifts for averaging */
628  w_ = 1 << jd->scale; /* Width of square */
629  op = workbuf;
630  iy = 0;
631  do {
632  ix = 0;
633  do {
634  rgb24 = &workbuf[(iy * mx + ix) * 3];
635  r_ = g_ = b_ = 0;
636  y_ = 0;
637  do { /* Accumulate RGB value in the square */
638  x_ = 0;
639  do {
640  r_ += rgb24[x_*3 ];
641  g_ += rgb24[x_*3+1];
642  b_ += rgb24[x_*3+2];
643  } while (++x_ < w_);
644  rgb24 += mx * 3;
645  } while (++y_ < w_);
646  /* Put the averaged RGB value as a pixel */
647  op[0] = r_ >> s_;
648  op[1] = g_ >> s_;
649  op[2] = b_ >> s_;
650  op += 3;
651  } while ((ix += w_) < mx);
652  } while ((iy += w_) < my);
653  }
654 
655  } else { /* For only 1/8 scaling (left-top pixel in each block are the DC value of the block) */
656 
657  /* Build a 1/8 descaled RGB MCU from discrete comopnents */
658  rgb24 = workbuf;
659  pc = jd->mcubuf + mx * my;
660  cb = pc[0] - 128; /* Get Cb/Cr component and restore right level */
661  cr = pc[64] - 128;
662  iy = 0;
663  do {
664  py = jd->mcubuf;
665  if (iy == 8) py += 64 * 2;
666  ix = 0;
667  do {
668  yy = *py; /* Get Y component */
669  py += 64;
670 
671  /* Convert YCbCr to RGB */
672  rgb24[0] = BYTECLIP(yy + (((int32_t)(1.402 * (1<<FP_SHIFT)) * cr) >> FP_SHIFT));
673  rgb24[1] = BYTECLIP(yy - (((int32_t)(0.34414 * (1<<FP_SHIFT)) * cb
674  + (int32_t)(0.71414 * (1<<FP_SHIFT)) * cr) >> FP_SHIFT));
675  rgb24[2] = BYTECLIP(yy + (((int32_t)(1.772 * (1<<FP_SHIFT)) * cb) >> FP_SHIFT));
676  rgb24 += 3;
677  } while ((ix += 8) < mx);
678  } while ((iy += 8) < my);
679  }
680 
681  /* Squeeze up pixel table if a part of MCU is to be truncated */
682  mx >>= jd->scale;
683  if (rx < mx) {
684  uint8_t *s_, *d;
685  s_ = d = workbuf;
686  for (size_t y_ = 1; y_ < ry; ++y_) {
687  memcpy(d += rx * 3, s_ += mx * 3, rx * 3); /* Copy effective pixels */
688  }
689  }
690 
691  /* Convert RGB888 to RGB565 if needed */
692  if (JD_FORMAT == 1) {
693  uint8_t *s = workbuf;
694  uint16_t *d = (uint16_t*)s;
695  uint_fast16_t w;
696  uint_fast16_t n = rx * ry;
697 
698  do {
699  w = (*s++ & 0xF8) << 8; /* RRRRR----------- */
700  w |= (*s++ & 0xFC) << 3; /* -----GGGGGG----- */
701  w |= *s++ >> 3; /* -----------BBBBB */
702  *d++ = w;
703  } while (--n);
704  }
705 
706  /* Output the RGB rectangular */
707  return outfunc(jd->device, workbuf, &rect) ? JDR_OK : JDR_INTR;
708 }

717 static JRESULT restart (
718  lgfxJdec* jd, /* Pointer to the decompressor object */
719  uint16_t rstn /* Expected restert sequense number */
720 )
721 {
722  uint16_t d;
723  uint8_t *dp = jd->dptr, *dpend = jd->dpend;
724 
725  /* Discard padding bits and get two bytes from the input stream */
726  d = 0;
727  for (int i = 0; i < 2; ++i) {
728  if (++dp == dpend) { /* No input data is available, re-fill input buffer */
729  dp = jd->inbuf;
730  jd->dpend = dpend = dp + jd->infunc(jd->device, dp, JD_SZBUF);
731  if (dp == dpend) return JDR_INP;
732  }
733  d = (d << 8) | *dp; /* Get a byte */
734  }
735  jd->dptr = dp; jd->dbit = 0;
736 
737  /* Check the marker */
738  if ((d & 0xFFD8) != 0xFFD0 || (d & 7) != (rstn & 7)) {
739  return JDR_FMT1; /* Err: expected RSTn marker is not detected (may be collapted data) */
740  }
741 
742  /* Reset DC offset */
743  jd->dcv[2] = jd->dcv[1] = jd->dcv[0] = 0;
744 
745  return JDR_OK;
746 }

762 JRESULT lgfx_jd_prepare (
763  lgfxJdec* jd, /* Blank decompressor object */
764  uint32_t (*infunc)(void*, uint8_t*, uint32_t), /* JPEG strem input function */
765  void* pool, /* Working buffer for the decompression session */
766  uint_fast16_t sz_pool, /* Size of working buffer */
767  void* dev /* I/O device identifier for the session */
768 )
769 {
770  uint8_t *seg;
771  uint32_t ofs;
772  size_t n;
773  int32_t rc;
774 
775 
776  if (!pool) return JDR_PAR;
777 
778  jd->pool = (uint8_t*)pool; /* Work memroy */
779  jd->sz_pool = sz_pool; /* Size of given work memory */
780  jd->infunc = infunc; /* Stream input function */
781  jd->device = dev; /* I/O device identifier */
782  jd->nrst = 0; /* No restart interval (default) */
783 
784 // memset(jd->huffbits, 0, sizeof(uint8_t*) * 4); /* Nulls pointers */
785 // memset(jd->huffcode, 0, sizeof(uint16_t*) * 4);
786 // memset(jd->huffdata, 0, sizeof(uint8_t*) * 4);
787 // memset(jd->qttbl, 0, sizeof(uint32_t*) * 4);
788 
789  jd->inbuf = seg = alloc_pool(jd, JD_SZBUF); /* Allocate stream input buffer */
790  if (!seg) return JDR_MEM1;
791 
792  if (infunc(dev, seg, 2) != 2) return JDR_INP;/* Check SOI marker */
793  if (LDB_WORD(seg) != 0xFFD8) return JDR_FMT1; /* Err: SOI is not detected */
794  ofs = 2;
795 
796  for (;;) {
797  if (infunc(dev, seg, 1) != 1) return JDR_INP;
798  if (seg[0] != 0xFF) return JDR_FMT1; /* Check a JPEG marker */
799  do
800  {
801  if (infunc(dev, &seg[1], 1) != 1) return JDR_INP;
802  } while (seg[1] == 0xFF);
803  if (infunc(dev, &seg[2], 2) != 2) return JDR_INP;
804  uint_fast16_t len = LDB_WORD(seg + 2) - 2; /* Length field */
805  ofs += 4 + len; /* Number of bytes loaded */
806 
807  switch (seg[1]) { /* Marker */
808  case 0xC0: /* SOF0 (baseline JPEG) */
809  {/* Load segment data */
810  if (len > JD_SZBUF) return JDR_MEM2;
811  if (infunc(dev, seg, len) != len) return JDR_INP;
812 
813  jd->width = LDB_WORD(seg+3); /* Image width in unit of pixel */
814  jd->height = LDB_WORD(seg+1); /* Image height in unit of pixel */
815  jd->comps_in_frame = seg[5];
816  if (seg[5] != 1 && seg[5] != 3)
817  return JDR_FMT3; /* Err: Supports only Y/Cb/Cr or Y(Grayscale) format */
818 
819  /* Check three image components */
820  for (size_t i = 0; i < seg[5]; ++i) {
821  uint_fast8_t b = seg[7 + 3 * i]; /* Get sampling factor */
822  if (!i) { /* Y component */
823  if (b != 0x11 && b != 0x22 && b != 0x21) { /* Check sampling factor */
824  return JDR_FMT3; /* Err: Supports only 4:4:4, 4:2:0 or 4:2:2 */
825  }
826  jd->msx = b >> 4; jd->msy = b & 15; /* Size of MCU [blocks] */
827  } else { /* Cb/Cr component */
828  if (b != 0x11) return JDR_FMT3; /* Err: Sampling factor of Cr/Cb must be 1 */
829  }
830  b = seg[8 + 3 * i]; /* Get dequantizer table ID for this component */
831  if (b > 3) return JDR_FMT3; /* Err: Invalid ID */
832  jd->qtid[i] = b;
833  }
834  }
835  break;
836 
837  case 0xDD: /* DRI */
838  {/* Load segment data */
839  if (len > JD_SZBUF) return JDR_MEM2;
840  if (infunc(dev, seg, len) != len) return JDR_INP;
841 
842  /* Get restart interval (MCUs) */
843  jd->nrst = LDB_WORD(seg);
844  }
845  break;
846 
847  case 0xC4: /* DHT */
848  {/* Load segment data */
849  if (len > JD_SZBUF) return JDR_MEM2;
850  if (infunc(dev, seg, len) != len) return JDR_INP;
851 
852  /* Create huffman tables */
853  rc = create_huffman_tbl(jd, seg, len);
854  if (rc) return (JRESULT)rc;
855  }
856  break;
857 
858  case 0xDB: /* DQT */
859  {/* Load segment data */
860  if (len > JD_SZBUF) return JDR_MEM2;
861  if (infunc(dev, seg, len) != len) return JDR_INP;
862 
863  /* Create de-quantizer tables */
864  rc = create_qt_tbl(jd, seg, len);
865  if (rc) return (JRESULT)rc;
866  }
867  break;
868 
869  case 0xDA: /* SOS */
870  {/* Load segment data */
871  if (len > JD_SZBUF) return JDR_MEM2;
872  if (infunc(dev, seg, len) != len) return JDR_INP;
873 
874  if (!jd->width || !jd->height) return JDR_FMT1; /* Err: Invalid image size */
875 
876  if (seg[0] != jd->comps_in_frame) return JDR_FMT3; /* Err: Supports only three color or grayscale components format */
877 
878  /* Check if all tables corresponding to each components have been loaded */
879  for (size_t i = 0; i < jd->comps_in_frame; ++i) {
880  uint_fast8_t b = seg[2 + 2 * i]; /* Get huffman table ID */
881  if (b != 0x00 && b != 0x11) return JDR_FMT3; /* Err: Different table number for DC/AC element */
882  b = i ? 1 : 0;
883  if (!jd->huffbits[b][0] || !jd->huffbits[b][1]) { /* Check dc/ac huffman table for this component */
884  return JDR_FMT1; /* Err: Nnot loaded */
885  }
886  if (!jd->qttbl[jd->qtid[i]]) { /* Check dequantizer table for this component */
887  return JDR_FMT1; /* Err: Not loaded */
888  }
889  }
890 
891  /* Allocate working buffer for MCU and RGB */
892  n = jd->msy * jd->msx; /* Number of Y blocks in the MCU */
893  if (!n) return JDR_FMT1; /* Err: SOF0 has not been loaded */
894  len = n * 64 * 2 + 64; /* Allocate buffer for IDCT and RGB output */
895  if (len < 256) len = 256; /* but at least 256 byte is required for IDCT */
896  jd->workbuf = alloc_pool(jd, len); /* and it may occupy a part of following MCU working buffer for RGB output */
897  if (!jd->workbuf) return JDR_MEM1; /* Err: not enough memory */
898  size_t mcubuf_len = (n + 2) * 64;
899  jd->mcubuf = (int16_t*)alloc_pool(jd, mcubuf_len * sizeof(int16_t)); /* Allocate MCU working buffer */
900  if (!jd->mcubuf) return JDR_MEM1; /* Err: not enough memory */
901  if (jd->comps_in_frame == 1) {
902  for (size_t i = n * 16; i < mcubuf_len; ++i) {
903  jd->mcubuf[i] = 128; /* Cb/Cr clear ( for grayscale )*/
904  }
905  }
906 
907  /* Pre-load the JPEG data to extract it from the bit stream */
908  ofs %= JD_SZBUF; /* Align read offset to JD_SZBUF */
909  int32_t dc = infunc(dev, seg + ofs, JD_SZBUF - ofs);
910  jd->dptr = seg + ofs - 1;
911  jd->dpend = seg + ofs + dc;
912  jd->dbit = 0; /* Prepare to read bit stream */
913 
914  }
915  return JDR_OK; /* Initialization succeeded. Ready to decompress the JPEG image. */
916 
917  case 0xC1: /* SOF1 */
918  case 0xC2: /* SOF2 */
919  case 0xC3: /* SOF3 */
920  case 0xC5: /* SOF5 */
921  case 0xC6: /* SOF6 */
922  case 0xC7: /* SOF7 */
923  case 0xC9: /* SOF9 */
924  case 0xCA: /* SOF10 */
925  case 0xCB: /* SOF11 */
926  case 0xCD: /* SOF13 */
927  case 0xCE: /* SOF14 */
928  case 0xCF: /* SOF15 */
929  case 0xD9: /* EOI */
930  return JDR_FMT3; /* Unsuppoted JPEG standard (may be progressive JPEG) */
931 
932  default: /* Unknown segment (comment, exif or etc..) */
933  /* Skip segment data */
934  if (infunc(dev, 0, len) != len) { /* Null pointer specifies to skip bytes of stream */
935  return JDR_INP;
936  }
937  }
938  }
