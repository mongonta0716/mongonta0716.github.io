563  }
564  }

568  const uint8_t *unicode_lut;
569 
570  font += this->start_pos_unicode();
571  unicode_lut = font;
572 
573  do

576  e = (pgm_read_byte(&unicode_lut[2]) << 8) + pgm_read_byte(&unicode_lut[3]);
577  unicode_lut += 4;
578  } while ( e < encoding );
579 
580  for ( ; 0 != (e = (pgm_read_byte(&font[0]) << 8) + pgm_read_byte(&font[1])) ; font += pgm_read_byte(&font[2]))
581  {
582  if ( e == encoding ) { return font + 3; } /* skip encoding and glyph size */
583  }
584  }
585  return nullptr;

614  size_t U8g2font::drawChar(LGFXBase* gfx, int32_t x, int32_t y, uint16_t uniCode, const TextStyle* style, FontMetrics* metrics, int32_t& filled_x) const
615  {
616  int32_t sy = 65536 * style->size_y;
617  y += (metrics->y_offset * sy) >> 16;
618  u8g2_font_decode_t decode(getGlyph(uniCode));
619  if ( decode.decode_ptr == nullptr ) return drawCharDummy(gfx, x, y, this->max_char_width(), metrics->height, style, filled_x);
620 
621  uint32_t w = decode.get_unsigned_bits(bits_per_char_width());
622  uint32_t h = decode.get_unsigned_bits(bits_per_char_height());
623 
624  int32_t sx = 65536 * style->size_x;
625 
626  int32_t xoffset = (decode.get_signed_bits(bits_per_char_x()) * sx) >> 16;
627 
628  int32_t yoffset = -(int32_t)(decode.get_signed_bits(bits_per_char_y()) + h + metrics->y_offset);
629 
630  int32_t xAdvance = (decode.get_signed_bits(bits_per_delta_x()) * sx) >> 16;
631 
632  uint32_t colortbl[2] = {gfx->getColorConverter()->convert(style->back_rgb888), gfx->getColorConverter()->convert(style->fore_rgb888)};
633  bool fillbg = (style->back_rgb888 != style->fore_rgb888);
634  int32_t left = 0;
635  int32_t right = 0;
636  if (fillbg) {
637  left = std::max<int>(filled_x, x + (xoffset < 0 ? xoffset : 0));
638  right = x + std::max<int>(((w * sx) >> 16) + xoffset, xAdvance);
639  filled_x = right;
640  gfx->setRawColor(colortbl[0]);
641  }
642  x += xoffset;
643  gfx->startWrite();
644 
645  if (left < right)
646  {
647  if (yoffset > 0) {
648  gfx->writeFillRect(left, y, right - left, (yoffset * sy) >> 16);
649  }
650  int32_t y0 = ((yoffset + h) * sy) >> 16;
651  int32_t y1 = (metrics->height * sy) >> 16;
652  if (y0 < y1) {
653  gfx->writeFillRect(left, y + y0, right - left, y1 - y0);
654  }
655  }
656 
657  if ( w > 0 )
658  {
659  if (left < right)
660  {
661  int32_t y0 = ( yoffset * sy) >> 16;
662  int32_t len = (((yoffset + h) * sy) >> 16) - y0;
663  if (left < x)
664  {
665  gfx->writeFillRect(left, y + y0, x - left, len);
666  }
667  int32_t xwsx = x + ((w * sx) >> 16);
668  if (xwsx < right)
669  {
670  gfx->writeFillRect(xwsx, y + y0, right - xwsx, len);
671  }
672  }
673  left -= x;
674  uint32_t ab[2];
675  uint32_t lx = 0;
676  uint32_t ly = 0;
677  int32_t y0 = ((yoffset ) * sy) >> 16;
678  int32_t y1 = ((yoffset + 1) * sy) >> 16;
679  do
680  {
681  ab[0] = decode.get_unsigned_bits(bits_per_0());
682  ab[1] = decode.get_unsigned_bits(bits_per_1());
683  bool i = 0;
684  do
685  {
686  uint32_t length = ab[i];
687  while (length)
688  {
689  uint32_t len = (length > w - lx) ? w - lx : length;
690  length -= len;
691  if (i || fillbg)
692  {
693  int32_t x0 = (lx * sx) >> 16;
694  if (!i && x0 < left) x0 = left;
695  int32_t x1 = ((lx + len) * sx) >> 16;
696  if (x0 < x1)
697  {
698  gfx->setRawColor(colortbl[i]);
699  gfx->writeFillRect( x + x0
700  , y + y0
701  , x1 - x0
702  , y1 - y0);
703  }
704  }
705  lx += len;
706  if (lx == w)
707  {
708  lx = 0;
709  ++ly;
710  y0 = y1;
711  y1 = ((ly + yoffset + 1) * sy) >> 16;
712  }
713  }
714  i = !i;
715  } while (i || decode.get_unsigned_bits(1) != 0 );
716  } while (ly < h);
717  }
718  gfx->endWrite();
719  return xAdvance;
720  }

791  bool VLWfont::loadFont(DataWrapper* data) {
792  _fontData = data;
793  {
794  uint32_t buf[6];
795  data->read((uint8_t*)buf, 6 * 4); // 24 Byte read
796 
797  gCount = getSwap32(buf[0]); // glyph count in file
798  //getSwap32(buf[1]); // vlw encoder version - discard
799  yAdvance = getSwap32(buf[2]); // Font size in points, not pixels
800  //getSwap32(buf[3]); // discard
801  ascent = getSwap32(buf[4]); // top of "d"
802  descent = getSwap32(buf[5]); // bottom of "p"
803  }
804 
805  // These next gFont values might be updated when the Metrics are fetched
806  maxAscent = ascent; // Determined from metrics
807  maxDescent = descent; // Determined from metrics
808  yAdvance = std::max((int)yAdvance, ascent + descent);
809  spaceWidth = yAdvance * 2 / 7; // Guess at space width
810 
811 //ESP_LOGI("LGFX", "ascent:%d descent:%d", gFont.ascent, gFont.descent);
812 
813  if (!gCount) return false;
814 
815 //ESP_LOGI("LGFX", "font count:%d", gCount);
816 
817  uint32_t bitmapPtr = 24 + (uint32_t)gCount * 28;
818 
819  gBitmap = (uint32_t*)heap_alloc_psram( gCount * 4); // seek pointer to glyph bitmap in the file
820  gUnicode = (uint16_t*)heap_alloc_psram( gCount * 2); // Unicode 16 bit Basic Multilingual Plane (0-FFFF)
821  gWidth = (uint8_t*)heap_alloc_psram( gCount ); // Width of glyph
822  gxAdvance = (uint8_t*)heap_alloc_psram( gCount ); // xAdvance - to move x cursor
823  gdX = (int8_t*)heap_alloc_psram( gCount ); // offset for bitmap left edge relative to cursor X
824 
825  if (nullptr == gBitmap ) gBitmap = (uint32_t*)heap_alloc( gCount * 4); // seek pointer to glyph bitmap in the file
826  if (nullptr == gUnicode ) gUnicode = (uint16_t*)heap_alloc( gCount * 2); // Unicode 16 bit Basic Multilingual Plane (0-FFFF)
827  if (nullptr == gWidth ) gWidth = (uint8_t*)heap_alloc( gCount ); // Width of glyph
828  if (nullptr == gxAdvance) gxAdvance = (uint8_t*)heap_alloc( gCount ); // xAdvance - to move x cursor
829  if (nullptr == gdX ) gdX = (int8_t*)heap_alloc( gCount ); // offset for bitmap left edge relative to cursor X
830 
831  if (!gUnicode
832  || !gBitmap
833  || !gWidth
834  || !gxAdvance
835  || !gdX) {
836 //ESP_LOGE("LGFX", "can not alloc font table");
837  return false;
838  }
839 
840  _fontLoaded = true;
841 
842  size_t gNum = 0;
843  _fontData->seek(24); // headerPtr
844  uint32_t buffer[7];
845  do {
846  _fontData->read((uint8_t*)buffer, 7 * 4); // 28 Byte read
847  uint16_t unicode = getSwap32(buffer[0]); // Unicode code point value
848  uint32_t w = (uint8_t)getSwap32(buffer[2]); // Width of glyph
849  if (gUnicode) gUnicode[gNum] = unicode;
850  if (gWidth) gWidth[gNum] = w;
851  if (gxAdvance) gxAdvance[gNum] = (uint8_t)getSwap32(buffer[3]); // xAdvance - to move x cursor
852  if (gdX) gdX[gNum] = (int8_t)getSwap32(buffer[5]); // x delta from cursor
853 
854  uint16_t height = getSwap32(buffer[1]); // Height of glyph
855  if ((unicode > 0xFF) || ((unicode > 0x20) && (unicode < 0xA0) && (unicode != 0x7F))) {
856  int16_t dY = (int16_t)getSwap32(buffer[4]); // y delta from baseline
857 //Serial.printf("LGFX:unicode:%x dY:%d\r\n", unicode, dY);
858  if (maxAscent < dY && unicode != 0x3000) {
859  maxAscent = dY;
860  }
861  if (maxDescent < (height - dY) && unicode != 0x3000) {
862 //Serial.printf("LGFX:maxDescent:%d\r\n", maxDescent);
863  maxDescent = height - dY;
864  }
865  }
866 
867  if (gBitmap) gBitmap[gNum] = bitmapPtr;
868  bitmapPtr += w * height;
869  } while (++gNum < gCount);
870 
871  yAdvance = maxAscent + maxDescent;
872 
873 //Serial.printf("LGFX:maxDescent:%d\r\n", maxDescent);
874  return true;
875  }

879  size_t VLWfont::drawChar(LGFXBase* gfx, int32_t x, int32_t y, uint16_t code, const TextStyle* style, FontMetrics* metrics, int32_t& filled_x) const
880  {
881  auto file = this->_fontData;
882 
883  uint32_t buffer[6] = {0};
884  uint16_t gNum = 0;
885 
886  int32_t sy = 65536 * style->size_y;
887  y += (metrics->y_offset * sy) >> 16;
888 
889  if (code == 0x20) {
890  gNum = 0xFFFF;
891  buffer[2] = getSwap32(this->spaceWidth);
892  } else if (!this->getUnicodeIndex(code, &gNum)) {
893  return drawCharDummy(gfx, x, y, this->spaceWidth, metrics->height, style, filled_x);
894  } else {
895  file->preRead();
896  file->seek(28 + gNum * 28);
897  file->read((uint8_t*)buffer, 24);
898  file->seek(this->gBitmap[gNum]);
899  }
900 
901 
902  int32_t h = getSwap32(buffer[0]); // Height of glyph
903  int32_t w = getSwap32(buffer[1]); // Width of glyph
904  int32_t sx = 65536 * style->size_x;
905  int32_t xAdvance = (getSwap32(buffer[2]) * sx) >> 16; // xAdvance - to move x cursor
906  int32_t xoffset = ((int32_t)((int8_t)getSwap32(buffer[4])) * sx) >> 16; // x delta from cursor
907  int32_t dY = (int16_t)getSwap32(buffer[3]); // y delta from baseline
908  int32_t yoffset = (this->maxAscent - dY);
909 // int32_t yoffset = (gfx->_font_metrics.y_offset) - dY;
910 
911  auto pixel = (uint8_t*)alloca(w * h);
912  if (gNum != 0xFFFF) {
913  file->read(pixel, w * h);
914  file->postRead();
915  }
916 
917  gfx->startWrite();
918 
919  uint32_t colortbl[2] = {gfx->getColorConverter()->convert(style->back_rgb888), gfx->getColorConverter()->convert(style->fore_rgb888)};
920  bool fillbg = (style->back_rgb888 != style->fore_rgb888);
921  int32_t left = 0;
922  int32_t right = 0;
923  if (fillbg) {
924  left = std::max(filled_x, x + (xoffset < 0 ? xoffset : 0));
925  right = x + std::max<int>(((w * sx) >> 16) + xoffset, xAdvance);
926  filled_x = right;
927  }
928  x += xoffset;
929 
930  int32_t clip_left;
931  int32_t clip_top;
932  int32_t clip_w;
933  int32_t clip_h;
934 
935  gfx->getClipRect(&clip_left, &clip_top, &clip_w, &clip_h);
936  int32_t clip_right = clip_left + clip_w-1;
937  int32_t clip_bottom = clip_top + clip_h-1;
938 
939  int32_t bx = x;
940  int32_t bw = (w * sx) >> 16;
941  if (x < clip_left) { bw += (x - clip_left); bx = clip_left; }
942 
943  if (bw > clip_right+1 - bx) bw = clip_right+1 - bx;
944 
945  if (bw >= 0)
946  {
947  int32_t fore_r = ((style->fore_rgb888>>16)&0xFF);
948  int32_t fore_g = ((style->fore_rgb888>> 8)&0xFF);
949  int32_t fore_b = ((style->fore_rgb888) &0xFF);
950 
951  if (fillbg || !gfx->isReadable() || gfx->hasPalette())
952  { // fill background mode or unreadable panel or palette sprite mode
953  if (left < right && fillbg) {
954  gfx->setRawColor(colortbl[0]);
955  if (yoffset > 0) {
956  gfx->writeFillRect(left, y, right - left, (yoffset * sy) >> 16);
957  }
958  int32_t y0 = ((yoffset + h) * sy) >> 16;
959  int32_t y1 = (metrics->height * sy) >> 16;
960  if (y0 < y1) {
961  gfx->writeFillRect(left, y + y0, right - left, y1 - y0);
962  }
963  }
964 
965  if (0 < w) {
966  uint32_t back = fillbg ? style->back_rgb888 : gfx->getBaseColor();
967  int32_t back_r = ((back>>16)&0xFF);
968  int32_t back_g = ((back>> 8)&0xFF);
969  int32_t back_b = ( back &0xFF);
970  int32_t i = 0;
971  int32_t y0, y1 = (yoffset * sy) >> 16;
972  do {
973  y0 = y1;
974  if (y0 > (clip_bottom - y)) break;
975  y1 = ((yoffset + i + 1) * sy) >> 16;
976  if (left < right) {
977  gfx->setRawColor(colortbl[0]);
978  gfx->writeFillRect(left, y + y0, right - left, y1 - y0);
979  }
980  int32_t j = 0;
981  do {
982  int32_t x0 = (j * sx) >> 16;
983  while (pixel[j] != 0xFF) {
984  int32_t x1 = ((j + 1) * sx) >> 16;
985  if (pixel[j] != 0 && x0 < x1) {
986  int32_t p = 1 + (uint32_t)pixel[j];

988  , ( fore_g * p + back_g * (257 - p)) >> 8
989  , ( fore_b * p + back_b * (257 - p)) >> 8 ));
990  gfx->writeFillRect(x + x0, y + y0, x1 - x0, y1 - y0);
991  }
992  x0 = x1;
993  if (++j == w || (clip_right - x) < x0) break;
994  }
995  if (j == w || (clip_right - x) < x0) break;
996  gfx->setRawColor(colortbl[1]);
997  do { ++j; } while (j != w && pixel[j] == 0xFF);
998  gfx->writeFillRect(x + x0, y + y0, ((j * sx) >> 16) - x0, y1 - y0);
999  } while (j != w);
1000  pixel += w;
1001  } while (++i < h);
1002  }
1003  }
1004  else // alpha blend mode
1005  {
1006  auto buf = (bgr888_t*)alloca((bw * ((sy + 65535) >> 16)) * sizeof(bgr888_t));
1007

1009  int32_t y0, y1 = (yoffset * sy) >> 16;
1010  int32_t i = 0;
1011  do {
1012  y0 = y1;
1013  if (y0 > (clip_bottom - y)) break;
1014  y1 = ((yoffset + i + 1) * sy) >> 16;
1015  int32_t by = y + y0;
1016  int32_t bh = y1 - y0;
1017 
1018  if (by < clip_top) { bh += by - clip_top; by = clip_top; }
1019  if (bh > 0) {
1020  int32_t j0 = 0;
1021 
1022  // search first and last pixel
1023  while (j0 != w && !pixel[j0 ]) { ++j0; }
1024  if (j0 != w) {
1025  int32_t j1 = w;
1026  while (j0 != j1 && !pixel[j1 - 1]) { --j1; }
1027  int32_t rx = (j0 * sx) >> 16;
1028  int32_t rw = (j1 * sx) >> 16;
1029  if (rx < bx -x) rx = bx -x;
1030  if (rw > bx+bw -x) rw = bx+bw -x;
1031  rw -= rx;
1032 
1033  if (0 < rw) {
1034  gfx->readRectRGB(x + rx, by, rw, bh, (uint8_t*)buf);
1035 
1036  int32_t x0, x1 = ((j0 * sx) >> 16) - rx;
1037  do {
1038  x0 = x1;
1039  if (x0 < 0) x0 = 0;
1040  x1 = (int)(((j0 + 1) * sx) >> 16) - rx;
1041  if (x1 > rw) x1 = rw;
1042  if (pixel[j0] && x0 < x1) {
1043  int32_t p = 1 + pixel[j0];
1044  do {
1045  int32_t yy = 0;
1046  do {
1047  auto bgr = &buf[x0 + yy * rw];
1048  bgr->r = ( fore_r * p + bgr->r * (257 - p)) >> 8;
1049  bgr->g = ( fore_g * p + bgr->g * (257 - p)) >> 8;
1050  bgr->b = ( fore_b * p + bgr->b * (257 - p)) >> 8;
1051  } while (++yy != bh);
1052  } while (++x0 != x1);
1053  }
1054  } while (++j0 < j1);
1055  gfx->pushImage(x + rx, by, rw, bh, &p_);
1056  }
1057  }
1058  }
1059  pixel += w;
1060  } while (++i < h);
1061  }
1062  }
1063  gfx->endWrite();
1064  return xAdvance;
