103  void Panel_HasBuffer::writeBlock(uint32_t rawcolor, uint32_t length)
104  {
105  uint32_t xs = _xs;
106  uint32_t xe = _xe;
107  uint32_t ys = _ys;
108  uint32_t ye = _ye;
109  if (xe < xs || ye < ys) { return; }
110  uint32_t xpos = _xpos;
111  uint32_t ypos = _ypos;
112  do
113  {
114  auto len = std::min<uint32_t>(length, xe + 1 - xpos);
115  writeFillRectPreclipped(xpos, ypos, len, 1, rawcolor);
116  xpos += len;
117  if (xpos > xe)
118  {
119  xpos = xs;
120  if (++ypos > ye)
121  {
122  ypos = ys;
123  }
124  }
125  length -= len;
126  } while (length);
127  _xs = xs;
