143  void Panel_Device::command_list(const uint8_t *addr)
144  {
145  for (;;)
146  { // For each command...
147  uint8_t cmd = *addr++;
148  uint8_t num = *addr++; // Number of args to follow
149  if (cmd == 0xFF && num == 0xFF) break;
150  writeCommand(cmd, 1); // Read, issue command
151  _bus->flush();
152  uint_fast8_t ms = num & CMD_INIT_DELAY; // If hibit set, delay follows args
153  num &= ~CMD_INIT_DELAY; // Mask out delay bit
154  if (num)
155  {
156  do
157  { // For each argument...
158  writeData(*addr++, 1); // Read, issue argument
159  _bus->flush();
160  } while (--num);
161  }
162  if (ms)
163  {
164  ms = *addr++; // Read post-command delay time (ms)
165  delay( (ms==255 ? 500 : ms) );
166  }
167  }
168  }

174  auto src_x = param->src_x;
175  auto buffer = reinterpret_cast<argb8888_t*>(const_cast<void*>(param->src_data));
176  auto bytes = param->dst_bits >> 3;
177 // ESP_LOGI("LGFX","DEBUG: %d %d", param->dst_bits, bytes);
178  // uint8_t* dmabuf = _bus->getFlipBuffer(w * bytes);

200 
201  pc_write.src_x = drawed_x;
202  writeImage(x + drawed_x, y, xstart - drawed_x, 1, &pc_write, true);
203  }
204  drawed_x = xstart + 1;
205  }
206  else
207  {
208  while (255 == buffer[xstart].a && ++xstart != w);
209  if (xstart == w) break;
210  uint32_t j = xstart;
211  while (++j != w && buffer[j].a && buffer[j].a != 255);
212  readRect(x + xstart, y, j - xstart + 1, 1, &dmabuf[xstart * bytes], &pc_read);

220

349  void Panel_Device::setCalibrate(uint16_t *parameters)
350  {
351  uint32_t vect[6] = {0,0,0,0,0,0};
352  float mat[3][3] = {{0,0,0},{0,0,0},{0,0,0}};
353  float a;
354 
355  auto w = _cfg.panel_width;
356  auto h = _cfg.panel_height;
357  if (_touch->config().offset_rotation & 1)
358  {
359  std::swap(w, h);
360  }
361 
362  for ( int i = 0; i < 4; ++i ) {
363  int32_t tx = (i & 2) ? (w - 1) : 0;
364  int32_t ty = (i & 1) ? (h - 1) : 0;
365  int32_t px = parameters[i*2 ];
366  int32_t py = parameters[i*2+1];
367  a = px * px;
368  mat[0][0] += a;
369  a = px * py;
370  mat[0][1] += a;
371  mat[1][0] += a;
372  a = px;
373  mat[0][2] += a;
374  mat[2][0] += a;
375  a = py * py;
376  mat[1][1] += a;
377  a = py;
378  mat[1][2] += a;
379  mat[2][1] += a;
380  mat[2][2] += 1;
381 
382  vect[0] += px * tx;
383  vect[1] += py * tx;
384  vect[2] += tx;
385  vect[3] += px * ty;
386  vect[4] += py * ty;
387  vect[5] += ty;
388  }
389 
390  {
391  for ( int k = 0; k < 3; ++k )
392  {
393  float t = mat[k][k];
394  for ( int i = 0; i < 3; ++i ) mat[k][i] /= t;
395 
396  mat[k][k] = 1 / t;
397  for ( int j = 0; j < 3; ++j )
398  {
399  if ( j == k ) continue;
400 
401  float u = mat[j][k];
402 
403  for ( int i = 0; i < 3; ++i )
404  {
405  if ( i != k ) mat[j][i] -= mat[k][i] * u;
406  else mat[j][i] = -u / t;
407  }
408  }
409  }
410  }
411 
412  float v0 = vect[0];
413  float v1 = vect[1];
414  float v2 = vect[2];
415  _affine[0] = mat[0][0] * v0 + mat[0][1] * v1 + mat[0][2] * v2;
416  _affine[1] = mat[1][0] * v0 + mat[1][1] * v1 + mat[1][2] * v2;
417  _affine[2] = mat[2][0] * v0 + mat[2][1] * v1 + mat[2][2] * v2;
418  float v3 = vect[3];
419  float v4 = vect[4];
420  float v5 = vect[5];
421  _affine[3] = mat[0][0] * v3 + mat[0][1] * v4 + mat[0][2] * v5;
422  _affine[4] = mat[1][0] * v3 + mat[1][1] * v4 + mat[1][2] * v5;
423  _affine[5] = mat[2][0] * v3 + mat[2][1] * v4 + mat[2][2] * v5;
424  }

426  void Panel_Device::convertRawXY(touch_point_t *tp, uint_fast8_t count)
427  {
428  auto r = _internal_rotation;
429  if (_touch) {
430  auto offset = _touch->config().offset_rotation;
431  r = ((r + offset) & 3) | ((r & 4) ^ (offset & 4));
432  }
433  bool vflip = (1 << r) & 0b10010110; // r 1,2,4,7
434 
435  for (size_t idx = 0; idx < count; ++idx)
436  {
437  int32_t tx = (_affine[0] * (float)tp[idx].x + _affine[1] * (float)tp[idx].y) + _affine[2];
438  int32_t ty = (_affine[3] * (float)tp[idx].x + _affine[4] * (float)tp[idx].y) + _affine[5];
439  if (r)
440  {
441  if (r & 1) { std::swap(tx, ty); }
442  if (r & 2) { tx = (_width - 1) - tx; }
443  if (vflip) { ty = (_height - 1) - ty; }
444  }
445  tp[idx].x = tx;
446  tp[idx].y = ty;
447  }
448  }
