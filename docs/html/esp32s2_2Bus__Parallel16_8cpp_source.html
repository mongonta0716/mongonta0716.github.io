252  size_t Bus_Parallel16::_flush(size_t count, bool dc)
253  {
254  bool slow = _div_num > 8;
255 
256  auto i2s_dev = (i2s_dev_t*)_dev;
257  if (i2s_dev->out_link.val)
258  {
259 #if defined (CONFIG_IDF_TARGET_ESP32S2)
260  while (!(i2s_dev->lc_state0.out_empty)) {}
261 #else
262  while (!(i2s_dev->lc_state0 & 0x80000000)) {} // I2S_OUT_EMPTY
263 #endif
264  }
265  _dmadesc[0].buf = (uint8_t *)_cache_flip;
266  *(uint32_t*)&_dmadesc[0] = count << 2 | count << 14 | 0xC0000000;
267  _dmadesc[0].empty = 0;
268 
269  while (!i2s_dev->state.tx_idle) {}
270 
271  i2s_dev->conf.val = _conf_reg_reset;
272  i2s_dev->sample_rate_conf.val = _sample_rate_conf_reg_bufferd;
273  i2s_dev->out_link.val = I2S_OUTLINK_START | ((uint32_t)&_dmadesc[0] & I2S_OUTLINK_ADDR);
274 
275  int wait = 40 - (_div_num << 3);
276  if (_direct_dc)
277  {
278  _direct_dc = false;
279  gpio_matrix_out(_cfg.pin_rs, I2S0O_DATA_OUT0_IDX, 0, 0);
280  wait -= 16;
281  }
282  if (wait > 0)
283  {
284  do { __asm__ __volatile__ ("nop"); } while (--wait);
285  }
286  i2s_dev->conf.val = _conf_reg_start;
287  if (slow)
288  {
289  wait = _div_num >> 1;
290  do { __asm__ __volatile__ ("nop"); } while (--wait);
291  }
292 
293  _cache_flip = (_cache_flip == _cache[0]) ? _cache[1] : _cache[0];
294  return 0;
295  }

485  {

505  _setup_dma_desc_links(data, len);
506  length -= len;
507  data += len;
508  }
509  else
510  {
511  size_t len = ((length - 1) % (CACHE_THRESH * sizeof(_cache[0][0]))) + 1;

518  data += len;

522  c[0] = _align_data;

524  }

526 
527  _cache_flip = (_cache_flip == _cache[0]) ? _cache[1] : _cache[0];
528  c = (uint8_t*)_cache_flip;
529  }

531  int wait = 40 - (_div_num << 3);
532  if (!_direct_dc)
533  {
534  _direct_dc = true;
535  gpio_matrix_out(_cfg.pin_rs, 0x100, 0, 0);
536  wait -= 16;
537  }

541  }
542  i2s_dev->conf.val = _conf_reg_start;
543  } while (length & ~1u);
544  }
545  if (length)
546  {

548  _align_data = data[0];
