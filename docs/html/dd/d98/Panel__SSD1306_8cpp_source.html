294  {

298  _range_mod.right = std::max<int_fast16_t>(_range_mod.right , x + w - 1);
299  _range_mod.top = std::min<int_fast16_t>(_range_mod.top , y );
300  _range_mod.bottom = std::max<int_fast16_t>(_range_mod.bottom, y + h - 1);
301  }
302  if (_range_mod.empty()) { return; }
303 
304  uint_fast8_t xs = _range_mod.left;
305  uint_fast8_t xe = _range_mod.right;
306  uint_fast8_t ys = _range_mod.top >> 3;
307  uint_fast8_t ye = _range_mod.bottom >> 3;
308  int retry = 3;
309  while (!(_bus->writeCommand(CMD_COLUMNADDR| (xs + _cfg.offset_x ) << 8 | (xe + _cfg.offset_x ) << 16, 24)
310  && _bus->writeCommand(CMD_PAGEADDR | (ys + (_cfg.offset_y >> 3)) << 8 | (ye + (_cfg.offset_y >> 3)) << 16, 24)) && --retry)
311  {
312  _bus->endTransaction();
313  _bus->beginTransaction();
314  }
315  if (retry)
316  {
317  do
318  {
319  auto buf = &_buf[xs + ys * _cfg.panel_width];
320  _bus->writeBytes(buf, xe - xs + 1, true, true);
321  } while (++ys <= ye);
322  _range_mod.top = INT16_MAX;
323  _range_mod.left = INT16_MAX;
324  _range_mod.right = 0;
