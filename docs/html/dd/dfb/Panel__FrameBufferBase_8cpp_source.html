139  void Panel_FrameBufferBase::writeBlock(uint32_t rawcolor, uint32_t length)
140  {
141  do
142  {
143  uint32_t h = 1;
144  auto w = std::min<uint32_t>(length, _xe + 1 - _xpos);
145  if (length >= (w << 1) && _xpos == _xs)
146  {
147  h = std::min<uint32_t>(length / w, _ye + 1 - _ypos);
148  }
149  writeFillRectPreclipped(_xpos, _ypos, w, h, rawcolor);
150  if ((_xpos += w) <= _xe) return;
151  _xpos = _xs;
152  if (_ye < (_ypos += h)) { _ypos = _ys; }
153  length -= w * h;
154  } while (length);

206  {
207  uint_fast16_t linelength;

212  cacheWriteBack(ptr, bytes * linelength);
213 
214  if ((x += linelength) > xe)
215  {
216  x = xs;
217  y = (y != ye) ? (y + 1) : ys;
218  }
219  } while (length -= linelength);

222  return;
223  }
224

226  int_fast16_t ay = 1;
227  if ((1u << r) & 0b10010110) { y = _height - (y + 1); ys = _height - (ys + 1); ye = _height - (ye + 1); ay = -1; }
228  if (r & 2) { x = _width - (x + 1); xs = _width - (xs + 1); xe = _width - (xe + 1); ax = -1; }
229 
230  if (r & 1)
231  {

235  if (x != xe)
236  {
237  x += ax;

242  y = (y != ye) ? (y + ay) : ys;

271  void Panel_FrameBufferBase::writeImage(uint_fast16_t x, uint_fast16_t y, uint_fast16_t w, uint_fast16_t h, pixelcopy_t* param, bool)
272  {
273  uint_fast8_t r = _internal_rotation;
274  if (r == 0 && param->transp == pixelcopy_t::NON_TRANSP && param->no_convert)
275  {
276  auto bits = _write_bits;
277  x = x * bits >> 3;
278  w = w * bits >> 3;
279  auto sw = param->src_bitwidth * bits >> 3;
280  auto src = &((uint8_t*)param->src_data)[param->src_y * sw + param->src_x];
281  h += y;
282  do
283  {
284  memcpy(&_lines_buffer[y][x], src, w);
285  cacheWriteBack(&_lines_buffer[y][x], w);
286  src += sw;
287  } while (++y != h);
288  return;
289  }
290 
291  uint32_t nextx = 0;
292  uint32_t nexty = 1 << pixelcopy_t::FP_SCALE;
293  if (r)
294  {
295  _rotate_pixelcopy(x, y, w, h, param, nextx, nexty);
296  }
297  uint32_t sx32 = param->src_x32;
298  uint32_t sy32 = param->src_y32;
299  uint_fast8_t bytes = _write_bits >> 3;
300  h += y;
301  do
302  {
303  int32_t pos = x;
304  int32_t end = pos + w;
305  while (end != (pos = param->fp_copy(_lines_buffer[y], pos, end, param))
306  && end != (pos = param->fp_skip( pos, end, param)));
307  param->src_x32 = (sx32 += nextx);
308  param->src_y32 = (sy32 += nexty);
309  auto ptr = &_lines_buffer[y][x * bytes];
310  cacheWriteBack(ptr, bytes * end);
311  } while (++y != h);

338  void Panel_FrameBufferBase::readRect(uint_fast16_t x, uint_fast16_t y, uint_fast16_t w, uint_fast16_t h, void* dst, pixelcopy_t* param)
339  {
340  uint_fast8_t r = _internal_rotation;
341  if (r == 0 && param->no_convert)
342  {
343  h += y;
344  auto bytes = _write_bits >> 3;
345  auto d = (uint8_t*)dst;
346  w *= bytes;
347  do
348  {
349  memcpy(d, &_lines_buffer[y][x * bytes], w);
350  d += w;
351  } while (++y != h);
352  return;
353  }
354 
355  int addx = 1;
356  int addy = 1;
357  uint_fast16_t wlen = 1;
358  if (r)
359  {
360  if (r & 2)
361  {
362  x = _width - (x + 1);
363  param->src_x32_add = -param->src_x32_add;
364  addx = -1;
365  }
366  if ((1 << r) & 0b10010110)
367  {
368  y = _height - (y + 1);
369  addy = -1;
370  }
371  if (r & 1)
372  {
373  std::swap(x, y);
374  std::swap(w, h);
375  std::swap(addx, addy);
376  std::swap(wlen, w);
377  std::swap(param->src_x32_add, param->src_y32_add);
378  }
379  }
380 
381  h = y + (h * addy);
382  uint_fast32_t pos = 0;
383  uint_fast16_t ybak = y;
384  do
385  {
386  y = ybak;
387  uint32_t x32 = x << pixelcopy_t::FP_SCALE;
388  do
389  {
390  param->src_y32 = 0;
391  param->src_x32 = x32;
392  param->src_data = _lines_buffer[y];
393  param->fp_copy(dst, pos, pos + w, param);
394  pos += w;
395  } while (h != (y += addy));
396  x += addx;
397  } while (--wlen);
398  }
