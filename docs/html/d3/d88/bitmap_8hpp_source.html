68  static bool load_bmp_rle8(DataWrapper* data, uint8_t* linebuf, uint_fast16_t width)
69  {
70  width = (width + 3) & ~3;

72  uint_fast16_t xidx = 0;
73  bool eol = false;
74  do {

76  if (code[0] == 0) {
77  switch (code[1]) {

101  static bool load_bmp_rle4(DataWrapper* data, uint8_t* linebuf, uint_fast16_t width)
102  {
103  width = (width + 3) & ~3;
104  uint8_t code[2];
105  uint_fast16_t xidx = 0;
106  bool eol = false;
107  do {
108  data->read(code, 2);
109  if (code[0] == 0) {
110  switch (code[1]) {
111  case 0x00: // EOL
112  case 0x01: // EOB
113  eol = true;
114  break;
115 
116  case 0x02: // move info (not support)
117  return false;
118 
119  default: // 絶対モードデータ;
120  {
121  int_fast16_t len = code[1];
122  int_fast16_t dbyte = ((int_fast16_t)code[1] + 1) >> 1;
123 
124  data->read(&linebuf[(xidx + 1) >> 1], (dbyte + 1) & ~1); // word align
125  if (xidx & 1) {
126  linebuf[xidx >> 1] |= linebuf[(xidx >> 1) + 1] >> 4;
127  for (long i = 1; i < dbyte; ++i) {
128  linebuf[((xidx + i) >> 1)] = (linebuf[((xidx + i) >> 1) ] << 4)
129  | linebuf[((xidx + i) >> 1) + 1] >> 4;
130  }
131  }
132  xidx += len;
133  }
134  break;
135  }
136  } else if (xidx + code[0] <= width) {
137  if (xidx & 1) {

139  code[1] = (code[1] >> 4) | (code[1] << 4);

144  } else {
