106  size_t Touch_FT5x06::_read_data(uint8_t* readdata)
107  {
109  size_t res = 0;
110  if (lgfx::i2c::beginTransaction(_cfg.i2c_port, _cfg.i2c_addr, _cfg.freq, false))
111  {
112  readdata[0] = 0x02;
113  if (lgfx::i2c::writeBytes(_cfg.i2c_port, readdata, 1)
114  && lgfx::i2c::restart(_cfg.i2c_port, _cfg.i2c_addr, _cfg.freq, true)
115  && lgfx::i2c::readBytes(_cfg.i2c_port, readdata, 1))
116  {
117  uint_fast8_t points = std::min<uint_fast8_t>(max_touch_points, readdata[0] & 0x0Fu);
118  if (points)
119  {
120  if (lgfx::i2c::readBytes(_cfg.i2c_port, &readdata[1], points * 6 - 2))
121  {
122  res = points * 6 - 1;
123  }
124  }
125  else
126  {
127  res = 1;
128  }
129  }
130  lgfx::i2c::endTransaction(_cfg.i2c_port).has_value();
131  }
132  return res;

135  uint_fast8_t Touch_FT5x06::getTouchRaw(touch_point_t *tp, uint_fast8_t count)
136  {
137  if (!_check_init() || count == 0) return 0;
138  if (_cfg.pin_int >= 0)
139  {
140  if (_flg_released != (bool)gpio_in(_cfg.pin_int))
141  {
142  _flg_released = !_flg_released;
143  _write_reg(FT5x06_INTMODE_REG, 0x00); // INT Polling mode
144  }
145  if (_flg_released)
146  {
147  return 0;
148  }
149  }
150  if (count > max_touch_points) { count = max_touch_points; }
151 
152  uint8_t readdata[2][30];
153  size_t readlen[2];
154  if (1 == (readlen[0] = _read_data(readdata[0]))) { return 0; }
155  size_t comparelen;
156  int32_t retry = 5;
157  do
158  { // 読出し中に値が変わる事があるので、連続読出しして前回と同値でなければリトライする;
159  readlen[retry & 1] = _read_data(readdata[retry & 1]);
160  comparelen = std::min(readlen[0], readlen[1]);
161  } while ((0 == comparelen || memcmp(readdata[0], readdata[1], readlen[0])) && --retry);
162 
163  if (count > readdata[0][0]) count = readdata[0][0];
164  for (size_t idx = 0; idx < count; ++idx)
165  {
166  auto data = &readdata[0][idx * 6];
167  tp[idx].size = 1;
168  tp[idx].x = (data[1] & 0x0F) << 8 | data[2];
169  tp[idx].y = (data[3] & 0x0F) << 8 | data[4];
170  tp[idx].id = data[3] >> 4;
171  }
172  return count;
173  }
