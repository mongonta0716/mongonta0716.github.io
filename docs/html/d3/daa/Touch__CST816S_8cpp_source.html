83  lgfx::pinMode(_cfg.pin_int, pin_mode_t::input_pullup);
84  }

87  // 画面に触れていない時、I2C通信でCST816が見つからない事がある。

93  size_t Touch_CST816S::_read_data(uint8_t* readdata)
94  {
95  size_t res = 0;
96  if (lgfx::i2c::beginTransaction(_cfg.i2c_port, _cfg.i2c_addr, _cfg.freq, false))
97  {
98  readdata[0] = 0x02;
99  if (lgfx::i2c::writeBytes(_cfg.i2c_port, readdata, 1)
100  && lgfx::i2c::restart(_cfg.i2c_port, _cfg.i2c_addr, _cfg.freq, true)
101  && lgfx::i2c::readBytes(_cfg.i2c_port, readdata, 1))
102  {
103  uint_fast8_t points = std::min<uint_fast8_t>(max_touch_points, readdata[0] & 0x0Fu);
104  if (points)
105  {
106  if (lgfx::i2c::readBytes(_cfg.i2c_port, &readdata[1], points * 6 - 2))
107  {
108  res = points * 6 - 1;
109  }
110  }
111  else
112  {
113  res = 1;
114  }
115  }
116  lgfx::i2c::endTransaction(_cfg.i2c_port).has_value();
117  }
118  return res;
119  }

121  uint_fast8_t Touch_CST816S::getTouchRaw(touch_point_t *tp, uint_fast8_t count)
122  {
123  if (!_inited && !_check_init()) return 0;
124 
125  if (count > max_touch_points) { count = max_touch_points; }
126 
127  uint8_t readdata[8];
128  int retry = 3;
129  do
130  {
131  if (!_read_reg(0x02, readdata, 6))
132  {
133  return 0;
134  }
135  count = readdata[0];
136  } while ((count > 1 || readdata[5] != 0x10) && --retry);
137  if (retry && count)
138  {
139  // ESP_LOGV("CST816S", "%02x %02x %02x %02x %02x %02x %02x %02x %d %d",
140  // readdata[0], readdata[1], readdata[2], readdata[3],
141  // readdata[4], readdata[5], readdata[6], readdata[7], _inited, lgfx::gpio_in(_cfg.pin_int));
142  tp[0].id = 0;
143  tp[0].size = 1;
144  tp[0].x = readdata[2] | (readdata[1] & 0x0F) << 8;
145  tp[0].y = readdata[4] | (readdata[3] & 0x0F) << 8;
146  }
147  return count;
148  }
