313  void Bus_SPI::writeDataRepeat(uint32_t data, uint_fast8_t bit_length, uint32_t count)
314  {
315  auto spi_mosi_dlen_reg = _spi_mosi_dlen_reg;
316  auto spi_w0_reg = _spi_w0_reg;
317  auto spi_cmd_reg = _spi_cmd_reg;
318  auto gpio_reg_dc = _gpio_reg_dc[1];
319  auto mask_reg_dc = _mask_reg_dc;
320 #if defined ( CONFIG_IDF_TARGET ) && !defined ( CONFIG_IDF_TARGET_ESP32 )
321  auto dma = _clear_dma_reg;
322  if (dma) { _clear_dma_reg = nullptr; }
323 #endif
324  if (1 == count)
325  {
326  --bit_length;
327  while (*spi_cmd_reg & SPI_USR); // wait SPI
328 #if defined ( CONFIG_IDF_TARGET ) && !defined ( CONFIG_IDF_TARGET_ESP32 )
329  if (dma) { *dma = 0; }
330 #endif
331  *gpio_reg_dc = mask_reg_dc; // D/C high (data)
332  *spi_mosi_dlen_reg = bit_length; // set bitlength
333  *spi_w0_reg = data; // set data
334  *spi_cmd_reg = SPI_EXECUTE; // exec SPI
335  return;
336  }
337 
338  uint32_t regbuf0 = data | data << bit_length;
339  uint32_t regbuf1;
340  uint32_t regbuf2;
341  // make 12Bytes data.
342  bool bits24 = (bit_length == 24);
343  if (bits24) {
344  regbuf1 = regbuf0 >> 8 | regbuf0 << 16;
345  regbuf2 = regbuf0 >>16 | regbuf0 << 8;
346  } else {
347  if (bit_length == 8) { regbuf0 |= regbuf0 << 16; }
348  regbuf1 = regbuf0;
349  regbuf2 = regbuf0;
350  }
351 
352  uint32_t length = bit_length * count; // convert to bitlength.
353  uint32_t len = (length <= 96u) ? length : (length <= 144u) ? 48u : 96u; // 1st send length = max 12Byte (96bit).
354 
355  length -= len;
356  --len;
357 
358  while (*spi_cmd_reg & SPI_USR) {} // wait SPI
359 #if defined ( CONFIG_IDF_TARGET ) && !defined ( CONFIG_IDF_TARGET_ESP32 )
360  if (dma) { *dma = 0; }
361 #endif
362  *gpio_reg_dc = mask_reg_dc; // D/C high (data)
363  *spi_mosi_dlen_reg = len;
364  // copy to SPI buffer register
365  spi_w0_reg[0] = regbuf0;
366  spi_w0_reg[1] = regbuf1;
367  spi_w0_reg[2] = regbuf2;
368  *spi_cmd_reg = SPI_EXECUTE; // exec SPI
369  if (0 == length) return;
370 
371  uint32_t regbuf[7] = { regbuf0, regbuf1, regbuf2, regbuf0, regbuf1, regbuf2, regbuf0 } ;
372 
373  // copy to SPI buffer register
374  memcpy((void*)&spi_w0_reg[3], regbuf, 24);
375  memcpy((void*)&spi_w0_reg[9], regbuf, 28);
376 
377  // limit = 64Byte / depth_bytes;
378  // When 24bit color, 504 bits out of 512bit buffer are used.
379  // When 16bit color, it uses exactly 512 bytes. but, it behaves like a ring buffer, can specify a larger size.
380  uint32_t limit;
381  if (bits24)
382  {
383  limit = 504;
384  len = length % limit;
385  }
386  else
387  {
388 #if defined ( CONFIG_IDF_TARGET_ESP32 )
389  limit = (1 << 11);
390 #else
391  limit = (1 << 9);
392 #endif
393  len = length & (limit - 1);
394  }
395 
396  if (len)
397  {
398  length -= len;
399  --len;
400  while (*spi_cmd_reg & SPI_USR);
401  *spi_mosi_dlen_reg = len;
402  *spi_cmd_reg = SPI_EXECUTE;
403  if (0 == length) return;
404  }
405 
406  while (*spi_cmd_reg & SPI_USR);
407  *spi_mosi_dlen_reg = limit - 1;
408  *spi_cmd_reg = SPI_EXECUTE;
409  while (length -= limit)
410  {
411  while (*spi_cmd_reg & SPI_USR);
412  *spi_cmd_reg = SPI_EXECUTE;
413  }
414  }

416  void Bus_SPI::writePixels(pixelcopy_t* param, uint32_t length)
417  {
418  const uint8_t bytes = param->dst_bits >> 3;
419  if (_cfg.dma_channel)
420  {
421  uint32_t limit = (bytes == 2) ? 32 : 24;
422  uint32_t len;
423  do
424  {
425  len = (limit << 1) <= length ? limit : length;
426  if (limit <= 256) limit <<= 1;
427  auto dmabuf = _flip_buffer.getBuffer(len * bytes);
428  param->fp_copy(dmabuf, 0, len, param);
429  writeBytes(dmabuf, len * bytes, true, true);
430  } while (length -= len);
431  return;
432  }
433 
435 #if defined ( SPI_UPDATE ) // for C3/S3
436 
437  const uint32_t limit = (bytes == 2) ? 32 : 21;
438  uint32_t l = (length - 1) / limit;
439  uint32_t len = length - (l * limit);
440  length = l;
441  uint32_t regbuf[16];
442  param->fp_copy(regbuf, 0, len, param);
443 
444  auto spi_w0_reg = _spi_w0_reg;
445 
446  dc_control(true);
447  set_write_len(len * bytes << 3);
448 
449  memcpy((void*)spi_w0_reg, regbuf, (len * bytes + 3) & (~3));
450 
451  exec_spi();
452  if (0 == length) return;
453 
454 
455  param->fp_copy(regbuf, 0, limit, param);
456  wait_spi();
457  set_write_len(limit * bytes << 3);
458  memcpy((void*)spi_w0_reg, regbuf, limit * bytes);
459  exec_spi();
460 
461 
462  while (--length)
463  {
464  param->fp_copy(regbuf, 0, limit, param);
465  wait_spi();
466  memcpy((void*)spi_w0_reg, regbuf, limit * bytes);
467  exec_spi();
468  }
469 
470 #else
471 
472  const uint32_t limit = (bytes == 2) ? 16 : 10; // limit = 32/bytes (bytes==2 is 16 bytes==3 is 10)
473  uint32_t len = (length - 1) / limit;
474  uint32_t highpart = (len & 1) << 3;
475  len = length - (len * limit);
476  uint32_t regbuf[8];
477  param->fp_copy(regbuf, 0, len, param);
478 
479  auto spi_w0_reg = _spi_w0_reg;
480 
481  uint32_t user_reg = *_spi_user_reg;
482 
483  dc_control(true);
484  set_write_len(len * bytes << 3);
485 
486  memcpy((void*)&spi_w0_reg[highpart], regbuf, (len * bytes + 3) & (~3));
487  if (highpart) *_spi_user_reg = user_reg | SPI_USR_MOSI_HIGHPART;
488  exec_spi();
489  if (0 == (length -= len)) return;
490 
491  for (; length; length -= limit)
492  {
493  param->fp_copy(regbuf, 0, limit, param);
494  memcpy((void*)&spi_w0_reg[highpart ^= 0x08], regbuf, limit * bytes);
495  uint32_t user = user_reg;
496  if (highpart) user |= SPI_USR_MOSI_HIGHPART;
497  if (len != limit)
498  {
499  len = limit;
500  wait_spi();
501  set_write_len(limit * bytes << 3);
502  *_spi_user_reg = user;
503  exec_spi();
504  }
505  else
506  {
507  wait_spi();
508  *_spi_user_reg = user;
509  exec_spi();
510  }
511  }

517  void Bus_SPI::writeBytes(const uint8_t* data, uint32_t length, bool dc, bool use_dma)
518  {
519  if (length <= 64)
520  {
521  auto spi_w0_reg = _spi_w0_reg;
522  auto aligned_len = (length + 3) & (~3);
523  length <<= 3;
524  dc_control(dc);
525  set_write_len(length);
526  memcpy((void*)spi_w0_reg, data, aligned_len);
527  exec_spi();
528  return;
529  }
530 
531  if (_cfg.dma_channel)
532  {
533  if (false == use_dma && length < 1024)
534  {
535  use_dma = true;
536  auto buf = _flip_buffer.getBuffer(length);
537  memcpy(buf, data, length);
538  data = buf;
539  }
540  if (use_dma)
541  {
542  auto spi_dma_out_link_reg = _spi_dma_out_link_reg;
543  auto cmd = _spi_cmd_reg;
544  while (*cmd & SPI_USR) {}
545  *spi_dma_out_link_reg = 0;
546  _setup_dma_desc_links(data, length);
547 #if defined ( SOC_GDMA_SUPPORTED )
548  auto dma = reg(SPI_DMA_CONF_REG(_spi_port));
549  *dma = 0;
550  uint32_t len = ((length - 1) & ((SPI_MS_DATA_BITLEN)>>3)) + 1;
551  *spi_dma_out_link_reg = DMA_OUTLINK_START_CH0 | ((int)(&_dmadesc[0]) & 0xFFFFF);
552  *dma = SPI_DMA_TX_ENA;
553  _clear_dma_reg = dma;
554 #else
555  auto dma_conf_reg = reg(SPI_DMA_CONF_REG(_spi_port));
556  auto dma_conf = *dma_conf_reg & ~(SPI_OUT_DATA_BURST_EN | SPI_AHBM_RST | SPI_AHBM_FIFO_RST | SPI_OUT_RST);
557  *dma_conf_reg = dma_conf | SPI_AHBM_RST | SPI_AHBM_FIFO_RST | SPI_OUT_RST;
558 
559  // 送信長が4の倍数の場合のみバーストモードを使用する
560  // ※ 以下の3つの条件が揃うと、DMA転送の末尾付近でデータが化ける現象が起きる。
561  // 1.送信クロック80MHz (APBクロックと1:1)
562  // 2.DMAバースト読出し有効
563  // 3.送信データ長が4の倍数ではない (1Byte~3Byteの端数がある場合)
564  dma_conf |= (length & 3) ? (SPI_OUTDSCR_BURST_EN) : (SPI_OUTDSCR_BURST_EN | SPI_OUT_DATA_BURST_EN);
565 
566  *dma_conf_reg = dma_conf;
567  uint32_t len = length;
568  *spi_dma_out_link_reg = SPI_OUTLINK_START | ((int)(&_dmadesc[0]) & 0xFFFFF);
569  _clear_dma_reg = spi_dma_out_link_reg;
570 #endif
