53  void Panel_sdl::_event_proc(void)
54  {
55  SDL_Event event;
56  while (SDL_PollEvent(&event))
57  {
58  if ((event.type == SDL_KEYDOWN) || (event.type == SDL_KEYUP))
59  {
60  int gpio = -1;
61  switch (event.key.keysym.sym)
62  {
63  case SDLK_LEFT: gpio = 39; break;
64  case SDLK_DOWN: gpio = 38; break;
65  case SDLK_RIGHT: gpio = 37; break;
66  case SDLK_UP: gpio = 36; break;
67  default: continue;
68  }
69  if (event.type == SDL_KEYDOWN) {
70  gpio_lo(gpio);
71  } else {
72  gpio_hi(gpio);
73  }
74  }
75  else if (event.type == SDL_MOUSEBUTTONDOWN || event.type == SDL_MOUSEBUTTONUP || event.type == SDL_MOUSEMOTION)
76  {
77  auto mon = getMonitorByWindowID(event.button.windowID);
78  if (mon != nullptr)
79  {
80  int x, y, w, h;
81  SDL_GetWindowSize(mon->window, &w, &h);
82  SDL_GetMouseState(&x, &y);
83  mon->touch_x = x * mon->panel->config().panel_width / w;
84  mon->touch_y = y * mon->panel->config().panel_height / h;
85  if (event.type == SDL_MOUSEBUTTONDOWN && event.button.button == SDL_BUTTON_LEFT)
86  {
87  mon->touched = true;
88  }
89  if (event.type == SDL_MOUSEBUTTONUP && event.button.button == SDL_BUTTON_LEFT)
90  {
91  mon->touched = false;
92  }
93  }
94  }
95  else
96  if (event.type == SDL_WINDOWEVENT
97  || event.type == SDL_QUIT) {
98  auto monitor = getMonitorByWindowID(event.window.windowID);
99  if (monitor) {
100  if (event.window.event == SDL_WINDOWEVENT_RESIZED) {
101  monitor->panel->sdl_invalidate();
102  }
103  else
104  if (event.window.event == SDL_WINDOWEVENT_CLOSE) {
105  monitor->closing = true;
106  }
107  }
108  }
109  }
110  }

113  static int detectDebugger(bool* running)
114  {
115  uint32_t prev_ms = SDL_GetTicks();
116  do {
117  SDL_Delay(1);

121  if (ms - prev_ms > 64) { _in_step_exec = 1023; }

128  void Panel_sdl::_update_proc(void)
129  {
130  for (auto it = _list_monitor.begin(); it != _list_monitor.end(); )
131  {
132  if ((*it)->closing) {
133  SDL_DestroyTexture((*it)->texture);

343  void Panel_sdl::sdl_update(void)
344  {
345  if (monitor.renderer == nullptr)
346  {
347  sdl_create(&monitor);
348  }
349 
350  bool step_exec = _in_step_exec;
351 
352  if (_texupdate_counter != _modified_counter) {
353  pixelcopy_t pc(nullptr, color_depth_t::rgb888_3Byte, _write_depth, false);
354  if (_write_depth == rgb565_2Byte) {
355  pc.fp_copy = pixelcopy_t::copy_rgb_fast<bgr888_t, swap565_t>;
356  } else if (_write_depth == rgb888_3Byte) {
357  pc.fp_copy = pixelcopy_t::copy_rgb_fast<bgr888_t, bgr888_t>;
358  } else if (_write_depth == rgb332_1Byte) {
359  pc.fp_copy = pixelcopy_t::copy_rgb_fast<bgr888_t, rgb332_t>;
360  } else if (_write_depth == grayscale_8bit) {
361  pc.fp_copy = pixelcopy_t::copy_rgb_fast<bgr888_t, grayscale_t>;
362  }
363 
364  if (0 == SDL_LockMutex(_sdl_mutex))
365  {
366  _texupdate_counter = _modified_counter;
367  for (int y = 0; y < _cfg.panel_height; ++y)
368  {
369  pc.src_x32 = 0;
370  pc.src_data = _lines_buffer[y];
371  pc.fp_copy(&_texturebuf[y * _cfg.panel_width], 0, _cfg.panel_width, &pc);
372  }
373  SDL_UnlockMutex(_sdl_mutex);
374  SDL_UpdateTexture(monitor.texture, nullptr, _texturebuf, _cfg.panel_width * sizeof(rgb888_t));
375  }
376  }
377 
378  if (_invalidated || (_display_counter != _texupdate_counter))
379  {
380  SDL_RendererInfo info;
381  if (0 == SDL_GetRendererInfo(monitor.renderer, &info)) {
382  // ステップ実行中はVSYNCを待機しない
383  if (((bool)(info.flags & SDL_RENDERER_PRESENTVSYNC)) == step_exec)
384  {
385  SDL_RenderSetVSync(monitor.renderer, !step_exec);
386  }
387  }
388  SDL_RenderCopy(monitor.renderer, monitor.texture, nullptr, nullptr);
389  SDL_RenderPresent(monitor.renderer);
390  _display_counter = _texupdate_counter;
391  if (_invalidated) {
392  _invalidated = false;
393  SDL_RenderCopy(monitor.renderer, monitor.texture, nullptr, nullptr);
394  SDL_RenderPresent(monitor.renderer);
395  }
396  }
397  }
