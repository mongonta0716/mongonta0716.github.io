514  uint_fast8_t div_read = apbfreq / (_bus->getReadClock() + 1) + 1;
515 
516  uint32_t retry = 8;
517  do
518  {
519  // ESP_LOGI(TAG, "FREQ:%lu , %lu DIV_W:%lu , %lu", _bus->getClock(), _bus->getReadClock(), div_write, div_read);
520  uint32_t fpga_id = _read_fpga_id();
521  // 受信したIDの先頭が "HD" なら正常動作
522  if ((fpga_id & 0xFFFF) == ('H' | 'D' << 8))
523  {
524  break;
525  }
526 
527  if (fpga_id == 0 || fpga_id == ~0u)
528  { // MISOが変化しない場合、コマンドが正しく受理されていないと仮定し送信速度を下げる。
529  _bus->setClock(apbfreq / ++div_write);
530  }
531  else
532  { // 受信データの先頭が HD でない場合は受信速度を下げる。
533  _bus->setReadClock(apbfreq / ++div_read);
534  }
535  } while (--retry);
536 
537  if (retry == 0) {
538  ESP_LOGW(TAG, "read FPGA ID failed.");
539  return false;
540  }
541 
542  startWrite();
543  bool res = _init_resolution();
544  endWrite();
545 
546  ESP_LOGI(TAG, "Initialize HDMI transmitter...");
547  if (!driver.init() )
548  {
549  ESP_LOGW(TAG, "HDMI transmitter Initialize failed.");
550  return false;
551  }
552 
553  ESP_LOGI(TAG, "done.");
554  return res;
555  }

557  uint32_t getPllParams(Panel_M5HDMI::video_clock_t* vc, uint32_t target_clock) {
558 
559  static constexpr const uint32_t base_clock = 74250000;
560 
561  uint32_t fb_clock = base_clock;
562  uint32_t save_diff = ~0u;
563  uint32_t fb_div = 1;
564  uint32_t in_div = base_clock / (target_clock + 1);
565  if (in_div == 0) { in_div = 1; }
566  for (;;)
567  {
568  uint32_t tmp_clock = fb_clock / in_div;
569  uint32_t diff = abs((int32_t)target_clock - (int32_t)tmp_clock);
570 // ESP_LOGE("M5HDMI", "FB:%d IN:%d diff:%d", fb_div, in_div, diff);
571  if (save_diff > diff)
572  {
573  save_diff = diff;
574  vc->feedback_divider = fb_div;
575  vc->input_divider = in_div;
576  if (diff == 0) { break; }
577  }
578  if (target_clock < tmp_clock)
579  {
580  if (++in_div > 24) { break; }
581  }
582  else
583  {
584  if (++fb_div > 64) { break; }
585  fb_clock = base_clock * fb_div;
586  }
587  }
588 
589  uint32_t result = base_clock * vc->feedback_divider / vc->input_divider;
590 
591  save_diff = ~0u;
592  static constexpr const uint8_t odiv_tbl[] = { 2, 4, 8, 16, 32, 48, 64, 80, 96, 112, 128 };
593  static constexpr const int32_t vco_target = 800000000; // 800 MHz
594  for (auto odiv : odiv_tbl) {
595  uint32_t diff = abs((int32_t)(result * odiv) - vco_target);
596 // ESP_LOGE("M5HDMI", "DIFF:%d ODIV:%d", diff, odiv);
597  if (save_diff < diff) { break; }
598  save_diff = diff;
599  vc->output_divider = odiv;
600  }
601 
602  return result;
603  }

605  bool Panel_M5HDMI::_init_resolution(void)
606  {
607  video_clock_t vc;
608  int32_t OUTPUT_CLOCK = getPllParams(&vc, _pixel_clock);
609 
610  int32_t TOTAL_RESOLUTION = OUTPUT_CLOCK / _refresh_rate;
611 
612  int mem_width = _cfg.memory_width ;
613  int mem_height = _cfg.memory_height;
614 
615  uint32_t diff = ~0u;
616  int vert_total = mem_height + 9;
617  int hori_total = TOTAL_RESOLUTION / vert_total;
618  int hori_tmp = hori_total, vert_tmp = vert_total;
619  int hori_min = mem_width + 32 + (mem_width >> (1+_scale_w));
620  int hori_max = mem_width + 768 + (mem_width >> 3);
621  if (hori_tmp > hori_max) { hori_tmp = hori_max; }
622  for (;;)
623  {
624  int d1 = TOTAL_RESOLUTION - (hori_tmp * vert_tmp);
625  uint32_t diff_abs = abs(d1);
626  if (diff > diff_abs)
627  {
628  diff = diff_abs;
629  hori_total = hori_tmp;
630  vert_total = vert_tmp;
631  if (diff == 0) { break; }
632  }
633  if (d1 >= 0) { ++vert_tmp; }
634  else if (--hori_tmp < hori_min) { break; }
635  }
636 
637  bool res = (hori_total > hori_min);
638  if (!res)
639  { // If the blanking period is too small, it will not work properly.
640  hori_total = hori_min;
641  ESP_LOGE(TAG, "resolution error. out of range %dx%d %.2f Hz", mem_width, mem_height, (double)_refresh_rate);
642  }
643 
644  video_timing_t vt;
645 
646  vt.v.active = mem_height;
647  uint32_t remain = vert_total - mem_height;
648  int sync = 1;
649  remain -= sync;
650  int porch = remain >> 1;
651  remain -= porch;
652  vt.v.front_porch = porch;
653  vt.v.sync = sync;
654  vt.v.back_porch = remain;
655 
656  vt.h.active = mem_width;
657  remain = hori_total - mem_width;
658  sync = 24 + (remain >> 8);
659  remain -= sync;
660  porch = (remain * 131) >> 8;
661  remain -= porch;
662  vt.h.front_porch = porch;
663  vt.h.sync = sync;
664  vt.h.back_porch = remain;
665 
666  setVideoTiming(&vt);
667  setScaling(_scale_w, _scale_h);
668  _set_video_clock(&vc);
669 
670  {
671  ESP_LOGD(TAG, "PLL feedback_div:%d input_div:%d output_div:%d OUTPUT_CLOCK:%d", vc.feedback_divider, vc.input_divider, vc.output_divider, (int)OUTPUT_CLOCK);
672  ESP_LOGD(TAG, "logical resolution: w:%d h:%d", _cfg.panel_width, _cfg.panel_height);
673  ESP_LOGD(TAG, "scaling resolution: w:%d h:%d", _cfg.panel_width * _scale_w, _cfg.panel_height * _scale_h);
674  ESP_LOGD(TAG, " output resolution: w:%d h:%d", _cfg.memory_width, _cfg.memory_height);
675  ESP_LOGD(TAG, "video timing(Hori) total:%d active:%d frontporch:%d sync:%d backporch:%d", vt.h.active + vt.h.front_porch + vt.h.sync + vt.h.back_porch, vt.h.active, vt.h.front_porch, vt.h.sync, vt.h.back_porch);
676  ESP_LOGD(TAG, "video timing(Vert) total:%d active:%d frontporch:%d sync:%d backporch:%d", vt.v.active + vt.v.front_porch + vt.v.sync + vt.v.back_porch, vt.v.active, vt.v.front_porch, vt.v.sync, vt.v.back_porch);
677  }
678 
679  return res;
680  }

726  void Panel_M5HDMI::config_resolution( const config_resolution_t& cfg_reso )
727  {
728  static constexpr int SCALE_MAX = 16;
729  static constexpr int RANGE_MAX = 2048;
730 
731  uint_fast16_t logical_width = cfg_reso.logical_width;
732  uint_fast16_t logical_height = cfg_reso.logical_height;
733  float refresh_rate = cfg_reso.refresh_rate;
734  uint_fast16_t output_width = cfg_reso.output_width;
735  uint_fast16_t output_height = cfg_reso.output_height;
736  uint_fast8_t scale_w = cfg_reso.scale_w;
737  uint_fast8_t scale_h = cfg_reso.scale_h;
738  _pixel_clock = cfg_reso.pixel_clock;
739 
740  if (output_width)
741  {
742  if (output_width > RANGE_MAX) { output_width = RANGE_MAX; }
743  if (logical_width > output_width) { logical_width = output_width; }
744  }
745  if (output_height)
746  {
747  if (output_height > RANGE_MAX) { output_height = RANGE_MAX; }
748  if (logical_height > output_height) { logical_height = output_height; }
749  }
750  if (logical_width == 0)
751  {
752  if (logical_height == 0)
753  {
754  logical_width = 1280;
755  logical_height = 720;
756  }
757  else
758  {
759  logical_width = (logical_height << 4) / 9;
760  }
761  }
762  else
763  if (logical_height == 0)
764  {
765  logical_height = (logical_width * 9) >> 4;
766  }
767  if (logical_width > RANGE_MAX) { logical_width = RANGE_MAX; }
768  if (logical_height > RANGE_MAX) { logical_height = RANGE_MAX; }
769 
770  if (refresh_rate > 512.0f) { refresh_rate = 512.0f; }
771  else if (refresh_rate < 1.0f)
772  {
773  auto total = logical_width * logical_height;
774  if (total > 1843200) // over 1920x960
775  {
776  refresh_rate = 24.0f;
777  }
778  else
779  if (total > 1024000) // over 1280x800
780  {
781  refresh_rate = 30.0f;
782  }
783  else
784  {
785  refresh_rate = 60.0f;
786  }
787  }
788 
789  int limit = 55296000 / refresh_rate;
790  _refresh_rate = refresh_rate;
791 
792  if (output_width == 0 && output_height == 0 && scale_w == 0 && scale_h == 0)
793  {
794  scale_w = 1280 / logical_width;
795  scale_h = 720 / logical_height;
796  if ((scale_w > 16)
797  || (scale_h > 16)
798  || (limit != 1280 * 720)
799  || (scale_w * logical_width != 1280)
800  || (scale_h * logical_height != 720))
801  {
802  scale_w = 1;
803  scale_h = 1;
804  for (int scale = 2; scale <= SCALE_MAX; ++scale)
805  {
806  uint32_t scale_height = scale * logical_height;
807  uint32_t scale_width = scale * logical_width;
808  uint32_t total = scale_width * scale_height;
809  if (scale_width > 1920 || scale_height > 1920 || total > limit) { break; }
810  scale_w = scale;
811  scale_h = scale;
812  }
813  }
814  output_width = scale_w * logical_width;
815  output_height = scale_h * logical_height;
816  if (output_height & 1) { output_height++; }
817  if ((output_width & 1) && (scale_w & 1)) { output_width += scale_w; }
818  }
819  else
820  {
821  if (scale_h == 0)
822  {
823  scale_h = output_height / logical_height;
824  }
825  if (scale_h > SCALE_MAX) { scale_h = SCALE_MAX; }
826  while (logical_height * scale_h > output_height) { --scale_h; }
827 
828  if (scale_w == 0)
829  {
830  scale_w = output_width / logical_width;
831  }
832  uint32_t w = output_width / scale_w;
833  while (scale_w > SCALE_MAX || w * scale_w != output_width || logical_width * scale_w > output_width)
834  {
835  w = output_width / --scale_w;
836  }
837  }
838 
839  _scale_w = scale_w;
840  _scale_h = scale_h;
841  _cfg.memory_width = output_width ;
842  _cfg.memory_height = output_height ;
843  _cfg.panel_width = logical_width ;
844  _cfg.panel_height = logical_height;
845  _cfg.offset_x = (output_width / scale_w - logical_width ) >> 1;
846  _cfg.offset_y = (output_height / scale_h - logical_height) >> 1;
847  setRotation(getRotation());
848  }
