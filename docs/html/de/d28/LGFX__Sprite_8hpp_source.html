419  bool create_palette(void)
420  {
421  if (_write_conv.bits > 8) return false;
422 
423  deletePalette();
424 
425  size_t palettes = 1 << _write_conv.bits;
426  _palette.reset(palettes * sizeof(bgr888_t), AllocationSource::Normal);
427  if (!_palette) { return false; }
428 
429  if (!(_write_conv.depth & color_depth_t::has_palette))
430  {
431  auto depth = (color_depth_t)(_write_conv.bits | color_depth_t::has_palette);
432  _write_conv.setColorDepth(depth);
433  _read_conv = _write_conv;
434  _panel_sprite.setColorDepth(depth);
435  }
436  _palette_count = palettes;
437  return true;
438  }

448  bool create_from_bmp(DataWrapper* data)
449  {
450  bitmap_header_t bmpdata;
451 
452  if (!bmpdata.load_bmp_header(data)
453  || ( bmpdata.biCompression > 3)) {
454  return false;
455  }
456  uint32_t seekOffset = bmpdata.bfOffBits;
457  uint_fast16_t bpp = bmpdata.biBitCount; // 24 bcBitCount 24=RGB24bit
458  setColorDepth(bpp < 32 ? bpp : 24);
459  uint32_t w = bmpdata.biWidth;
460  int32_t h = bmpdata.biHeight; // bcHeight Image height (pixels)
461  if (!createSprite(w, h)) return false;
462 
463  //If the value of Height is positive, the image data is from bottom to top
464  //If the value of Height is negative, the image data is from top to bottom.
465  int32_t flow = (h < 0) ? 1 : -1;
466  int32_t y = 0;
467  if (h < 0) h = -h;
468  else y = h - 1;
469 
470  if (bpp <= 8) {
471  if (!_palette) createPalette();
472  uint_fast16_t palettecount = 1 << bpp;
473  argb8888_t *palette = (argb8888_t*)alloca(sizeof(argb8888_t*) * palettecount);
474  data->seek(bmpdata.biSize + 14);
475  data->read((uint8_t*)palette, (palettecount * sizeof(argb8888_t))); // load palette
476  for (uint_fast16_t i = 0; i < _palette_count; ++i)
477  {
478  _palette.img24()[i].set(color_convert<bgr888_t, argb8888_t>(palette[i].get()));
479  }
480  }
481 
482  data->seek(seekOffset);

492  } else

495  } else {
496  data->read(lineBuffer, buffersize);
497  }
498  memcpy(&_img8[y * bitwidth * bpp >> 3], lineBuffer, (w * bpp + 7) >> 3);
499  y += flow;
500  } while (--h);
501  } else if (bpp == 16) {
502  do {
503  data->read(lineBuffer, buffersize);
504  auto img = (uint16_t*)(&_img8[y * bitwidth * bpp >> 3]);
505  y += flow;
506  for (size_t i = 0; i < w; ++i)
507  {
508  img[i] = (lineBuffer[i << 1] << 8) + lineBuffer[(i << 1) + 1];
509  }
510  } while (--h);
511  } else if (bpp == 24) {
512  do {
513  data->read(lineBuffer, buffersize);
514  auto img = &_img8[y * bitwidth * bpp >> 3];
515  y += flow;
516  for (size_t i = 0; i < w; ++i) {
517  img[i * 3 ] = lineBuffer[i * 3 + 2];
518  img[i * 3 + 1] = lineBuffer[i * 3 + 1];
519  img[i * 3 + 2] = lineBuffer[i * 3 ];
520  }
521  } while (--h);
522  } else if (bpp == 32) {
523  do {
524  data->read(lineBuffer, buffersize);
525  auto img = &_img8[y * bitwidth * 3];
526  y += flow;
527  for (size_t i = 0; i < w; ++i) {
528  img[i * 3 ] = lineBuffer[(i << 2) + 2];
529  img[i * 3 + 1] = lineBuffer[(i << 2) + 1];
530  img[i * 3 + 2] = lineBuffer[(i << 2) + 0];
531  }
532  } while (--h);
533  }
534  return true;
