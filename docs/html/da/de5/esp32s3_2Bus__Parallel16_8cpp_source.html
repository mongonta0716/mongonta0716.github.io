239  void Bus_Parallel16::writeData(uint32_t data, uint_fast8_t bit_length)
240  {
241  auto bytes = bit_length >> 3;
242  auto dev = _dev;
243  auto reg_lcd_user = &(dev->lcd_user.val);
244  dev->lcd_misc.val = LCD_CAM_LCD_CD_IDLE_EDGE;
245  if (_has_align_data)
246  {
247  _has_align_data = false;
248  dev->lcd_cmd_val.val = _align_data | (data << 8);
249  while (*reg_lcd_user & LCD_CAM_LCD_START) {}
250  *reg_lcd_user = LCD_CAM_LCD_2BYTE_EN | LCD_CAM_LCD_CMD | LCD_CAM_LCD_UPDATE_REG | LCD_CAM_LCD_START;
251  if (--bytes == 0) { return; }
252  data >>= 8;
253  }
254 
255  if (bytes > 1)
256  {
257  dev->lcd_cmd_val.val = data;
258  if (bytes == 4)
259  {
260  while (*reg_lcd_user & LCD_CAM_LCD_START) {}
261  *reg_lcd_user = LCD_CAM_LCD_CMD_2_CYCLE_EN | LCD_CAM_LCD_2BYTE_EN | LCD_CAM_LCD_CMD | LCD_CAM_LCD_UPDATE_REG | LCD_CAM_LCD_START;
262  return;
263  }
264  while (*reg_lcd_user & LCD_CAM_LCD_START) {}
265  *reg_lcd_user = LCD_CAM_LCD_2BYTE_EN | LCD_CAM_LCD_CMD | LCD_CAM_LCD_UPDATE_REG | LCD_CAM_LCD_START;
266  if (bytes == 2) { return; }
267  data >>= 16;
268  }
269  _has_align_data = true;
270  _align_data = data;
271  }

288  regbuf2 = regbuf0 >>16 | regbuf0 << 8;

295 
296  dst[0] = regbuf0;
297  dst[1] = regbuf1;
298  dst[2] = regbuf2;
299  dst[3] = regbuf0;
300  dst[4] = regbuf1;
301  dst[5] = regbuf2;
302  dst[6] = regbuf0;
303  dst[7] = regbuf1;
304  dst[8] = regbuf2;
305 
306  uint32_t length = count * (bit_length >> 3);
307  if (length > 36)
308  {
309  memcpy((void*)&dst[ 9], dst, 36);
310  if (length > 72)
311  {
312  memcpy((void*)&dst[18], dst, 72);
313  if (length > 144)
314  {
315  memcpy((void*)&dst[36], dst, 108);
316  }
317  }
318  }
319  uint32_t len;
320  do
321  {
322  len = length;
323  if (length > 252u)
324  {
325  len = ((((length - 1) % 252) + 12) / 12) * 12;

355  void Bus_Parallel16::writeBytes(const uint8_t* data, uint32_t length, bool dc, bool use_dma)
356  {
357  auto dev = _dev;
358  auto reg_lcd_user = &(dev->lcd_user.val);
359  if ((length + _has_align_data) > 1)
360  {
361  do
362  {
363  if (length <= 4)
364  {
365  if (dc)
366  {
367  writeData(*(const uint32_t*)data, length << 3);
368  }
369  else
370  {
371  writeCommand(*(const uint32_t*)data, length << 3);
372  }
373  return;
374  }
375 
376  if (use_dma && !_has_align_data)
377  {
378  size_t len = length & ~1u;
379  while (*reg_lcd_user & LCD_CAM_LCD_START) {}
380  _setup_dma_desc_links(&data[4], len - 4);
381  *_dma_out_link_reg = DMA_OUTLINK_START_CH0 | ((uintptr_t)(&_dmadesc[0]) & 0xFFFFF);
382  dev->lcd_cmd_val.val = data[0] | data[1] << 8 | data[2] <<16 | data[3] << 24;
383  length -= len;
384  data += len;
385  }
386  else
387  {
388  size_t len = (length > CACHE_SIZE)
389  ? (((length - 1) % (CACHE_SIZE-_has_align_data)) + 4) & ~3u
390  : length;
391  if (_has_align_data != (bool)(len & 1))
392  {
393  if (++len > length) { len -= 2; }
394  }
395  length -= len;
396  auto c = (uint8_t*)_cache_flip;
397  while (*reg_lcd_user & LCD_CAM_LCD_START) {}
398  memcpy(&c[_has_align_data], data, (len + 3) & ~3u);
399  data += len;
400  if (_has_align_data)
401  {
402  _has_align_data = false;
403  c[0] = _align_data;
404  ++len;
405  }
406  _setup_dma_desc_links(&c[4], len - 4);
407  *_dma_out_link_reg = DMA_OUTLINK_START_CH0 | ((uintptr_t)(&_dmadesc[0]) & 0xFFFFF);
408  _cache_flip = _cache[(_cache_flip == _cache[0])];
409  dev->lcd_cmd_val.val = c[0] | c[1] << 8 | c[2] <<16 | c[3] << 24;
410  }
411  dev->lcd_misc.val = dc
412  ? LCD_CAM_LCD_CD_IDLE_EDGE
413  : (LCD_CAM_LCD_CD_IDLE_EDGE | LCD_CAM_LCD_CD_CMD_SET | LCD_CAM_LCD_CD_DATA_SET);
414 
415  while (*_dma_outstatus_reg & DMA_OUTFIFO_EMPTY_CH0 ) {}
416 
417  *reg_lcd_user = LCD_CAM_LCD_ALWAYS_OUT_EN | LCD_CAM_LCD_2BYTE_EN | LCD_CAM_LCD_CMD_2_CYCLE_EN | LCD_CAM_LCD_DOUT | LCD_CAM_LCD_CMD | LCD_CAM_LCD_UPDATE_REG | LCD_CAM_LCD_START;
418  } while (length & ~1u);
419  }
420  if (length)
421  {
422  _has_align_data = true;
423  _align_data = data[0];
424  }
425  }

443  void Bus_Parallel16::_read_bytes(uint8_t* dst, uint32_t length)
444  {
445  union
446  {
447  uint32_t in32[2];
448  uint8_t in[8];
449  };
450 
451  uint32_t mh = (((((((((((((((
452  (_cfg.pin_d8 & 7)) << 3)
453  + (_cfg.pin_d9 & 7)) << 3)
454  + (_cfg.pin_d10 & 7)) << 3)
455  + (_cfg.pin_d11 & 7)) << 3)
456  + (_cfg.pin_d12 & 7)) << 3)
457  + (_cfg.pin_d13 & 7)) << 3)
458  + (_cfg.pin_d14 & 7)) << 3)
459  + (_cfg.pin_d15 & 7))
460  ;
461 
462  uint32_t ih = (((((((((((((((
463  (_cfg.pin_d8 >> 3)) << 3)
464  + (_cfg.pin_d9 >> 3)) << 3)
465  + (_cfg.pin_d10 >> 3)) << 3)
466  + (_cfg.pin_d11 >> 3)) << 3)
467  + (_cfg.pin_d12 >> 3)) << 3)
468  + (_cfg.pin_d13 >> 3)) << 3)
469  + (_cfg.pin_d14 >> 3)) << 3)
470  + (_cfg.pin_d15 >> 3))
471  ;
472 
473  uint32_t ml = (((((((((((((((
474  (_cfg.pin_d0 & 7)) << 3)
475  + (_cfg.pin_d1 & 7)) << 3)
476  + (_cfg.pin_d2 & 7)) << 3)
477  + (_cfg.pin_d3 & 7)) << 3)
478  + (_cfg.pin_d4 & 7)) << 3)
479  + (_cfg.pin_d5 & 7)) << 3)
480  + (_cfg.pin_d6 & 7)) << 3)
481  + (_cfg.pin_d7 & 7))
482  ;
483 
484  uint32_t il = (((((((((((((((
485  (_cfg.pin_d0 >> 3)) << 3)
486  + (_cfg.pin_d1 >> 3)) << 3)
487  + (_cfg.pin_d2 >> 3)) << 3)
488  + (_cfg.pin_d3 >> 3)) << 3)
489  + (_cfg.pin_d4 >> 3)) << 3)
490  + (_cfg.pin_d5 >> 3)) << 3)
491  + (_cfg.pin_d6 >> 3)) << 3)
492  + (_cfg.pin_d7 >> 3))
493  ;
494 
495  auto reg_rd_h = get_gpio_hi_reg(_cfg.pin_rd);
496  auto reg_rd_l = get_gpio_lo_reg(_cfg.pin_rd);
497  uint32_t mask_rd = 1ul << (_cfg.pin_rd & 31);
498  uint32_t val;
499  do
500  {
501  in32[1] = GPIO.in1.val;
502  in32[0] = GPIO.in;
503  *reg_rd_h = mask_rd;
504  val = (1 & (in[(ih >> 0) & 7] >> ((mh >> 0) & 7)));
505  val = (val << 1) + (1 & (in[(ih >> 3) & 7] >> ((mh >> 3) & 7)));
506  val = (val << 1) + (1 & (in[(ih >> 6) & 7] >> ((mh >> 6) & 7)));
507  val = (val << 1) + (1 & (in[(ih >> 9) & 7] >> ((mh >> 9) & 7)));
508  val = (val << 1) + (1 & (in[(ih >> 12) & 7] >> ((mh >> 12) & 7)));
509  val = (val << 1) + (1 & (in[(ih >> 15) & 7] >> ((mh >> 15) & 7)));
510  val = (val << 1) + (1 & (in[(ih >> 18) & 7] >> ((mh >> 18) & 7)));
511  val = (val << 1) + (1 & (in[(ih >> 21) & 7] >> ((mh >> 21) & 7)));
512  *dst++ = val;
513  *reg_rd_l = mask_rd;
514 
515  if (0 == --length) { *reg_rd_l = mask_rd; break; }
516 
517  val = (1 & (in[(il >> 0) & 7] >> ((ml >> 0) & 7)));
518  val = (val << 1) + (1 & (in[(il >> 3) & 7] >> ((ml >> 3) & 7)));
519  val = (val << 1) + (1 & (in[(il >> 6) & 7] >> ((ml >> 6) & 7)));
520  val = (val << 1) + (1 & (in[(il >> 9) & 7] >> ((ml >> 9) & 7)));
521  val = (val << 1) + (1 & (in[(il >> 12) & 7] >> ((ml >> 12) & 7)));
522  val = (val << 1) + (1 & (in[(il >> 15) & 7] >> ((ml >> 15) & 7)));
523  val = (val << 1) + (1 & (in[(il >> 18) & 7] >> ((ml >> 18) & 7)));
524  val = (val << 1) + (1 & (in[(il >> 21) & 7] >> ((ml >> 21) & 7)));
525  *dst++ = val;
526  } while (--length);
527  }
