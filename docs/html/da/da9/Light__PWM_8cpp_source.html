84  void Light_PWM::setBrightness(uint8_t brightness)
85  {
86  uint32_t duty = 0;
87  if (brightness)
88  {
89  uint_fast16_t ofs = _cfg.offset;
90  if (ofs) { ofs = ofs * 259 >> 8; }
91  duty = brightness * (257 - ofs);
92  duty += ofs * 255;
93  duty += 1 << (15 - PWM_BITS);
94  duty >>= 16 - PWM_BITS;
95  }
96  if (_cfg.invert) duty = (1 << PWM_BITS) - duty;
97 
98 #if defined ( ARDUINO )
99  ledcWrite(_cfg.pwm_channel, duty);
100 #elif SOC_LEDC_SUPPORT_HS_MODE
101  ledc_set_duty(LEDC_HIGH_SPEED_MODE, (ledc_channel_t)_cfg.pwm_channel, duty);
102  ledc_update_duty(LEDC_HIGH_SPEED_MODE, (ledc_channel_t)_cfg.pwm_channel);
103 #else
104  ledc_set_duty(LEDC_LOW_SPEED_MODE, (ledc_channel_t)_cfg.pwm_channel, duty);
105  ledc_update_duty(LEDC_LOW_SPEED_MODE, (ledc_channel_t)_cfg.pwm_channel);
106 #endif
107  }
