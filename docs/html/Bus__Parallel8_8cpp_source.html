254 #endif

257  _dmadesc.buf = (uint8_t*)_cache_flip;

260  i2s_dev->conf.val = _conf_reg_reset | I2S_TX_FIFO_RESET;

497  void Bus_Parallel8::writeBytes(const uint8_t* data, uint32_t length, bool dc, bool use_dma)
498  {
499  uint32_t dc_data = dc ? 0x01000100 : 0;
500  auto idx = _cache_index;
501  auto c = _cache_flip;
502 
503  for (;;)
504  {
505  while (length && ((idx & 3) || (length < 4)))
506  {
507  --length;
508  c[idx^1] = *data++ | dc_data;
509  ++idx;
510  }
511  if (CACHE_THRESH <= idx)
512  {
513  idx = _flush(idx);
514  c = _cache_flip;
515  }
516  if (!length)
517  {
518  break;
519  }
520  size_t limit = std::min<uint32_t>(CACHE_THRESH, (length + idx) & ~3);
521  length -= (limit - idx);
522  do
523  {
524  *(uint32_t*)(&c[idx ]) = (data[0] << 16) | data[1] | dc_data;
525  *(uint32_t*)(&c[idx+2]) = (data[2] << 16) | data[3] | dc_data;
526  data += 4;
527  } while ((idx += 4) < limit);
528  }
529  _cache_index = idx;
