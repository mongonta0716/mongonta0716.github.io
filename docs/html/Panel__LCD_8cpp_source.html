124  void Panel_LCD::setRotation(uint_fast8_t r)
125  {
126  r &= 7;
127  _rotation = r;
128  // offset_rotationを加算 (0~3:回転方向、 4:上下反転フラグ);
129  _internal_rotation = ((r + _cfg.offset_rotation) & 3) | ((r & 4) ^ (_cfg.offset_rotation & 4));
130 
131  auto ox = _cfg.offset_x;
132  auto oy = _cfg.offset_y;
133  auto pw = _cfg.panel_width;
134  auto ph = _cfg.panel_height;
135  auto mw = _cfg.memory_width;
136  auto mh = _cfg.memory_height;
137  if (_internal_rotation & 1)
138  {
139  std::swap(ox, oy);
140  std::swap(pw, ph);
141  std::swap(mw, mh);
142  }
143  _width = pw;
144  _height = ph;
145  _colstart = (_internal_rotation & 2)
146  ? mw - (pw + ox) : ox;
147 
148  _rowstart = ((1 << _internal_rotation) & 0b10010110) // case 1:2:4:7
149  ? mh - (ph + oy) : oy;
150 
151  _xs = _xe = _ys = _ye = INT16_MAX;

286  void Panel_LCD::writeImage(uint_fast16_t x, uint_fast16_t y, uint_fast16_t w, uint_fast16_t h, pixelcopy_t* param, bool use_dma)
287  {
288  auto bytes = param->dst_bits >> 3;
289  auto src_x = param->src_x;
290 
291  if (param->transp == pixelcopy_t::NON_TRANSP)
292  {
293  if (param->no_convert)
294  {
295  auto wb = w * bytes;
296  uint32_t i = (src_x + param->src_y * param->src_bitwidth) * bytes;
297  auto src = &((const uint8_t*)param->src_data)[i];
298  setWindow(x, y, x + w - 1, y + h - 1);
299  if (param->src_bitwidth == w || h == 1)
300  {
301  write_bytes(src, wb * h, use_dma);
302  }
303  else
304  {
305  auto add = param->src_bitwidth * bytes;
306  if (use_dma)
307  {
308  if (_cfg.dlen_16bit && ((wb * h) & 1))
309  {
310  _has_align_data = !_has_align_data;
311  }
312  do
313  {
314  _bus->addDMAQueue(src, wb);
315  src += add;
316  } while (--h);
317  _bus->execDMAQueue();
318  }
319  else
320  {
321  do
322  {
323  write_bytes(src, wb, false);
324  src += add;
325  } while (--h);
326  }
327  }
328  }
329  else
330  {
331  if (!_bus->busy())
332  {
333  static constexpr uint32_t WRITEPIXELS_MAXLEN = 32767;
334 
335  setWindow(x, y, x + w - 1, y + h - 1);
336  // bool nogap = (param->src_bitwidth == w || h == 1);
337  bool nogap = (h == 1) || (param->src_y32_add == 0 && ((param->src_bitwidth << pixelcopy_t::FP_SCALE) == (w * param->src_x32_add)));
338  if (nogap && (w * h <= WRITEPIXELS_MAXLEN))
339  {
340  writePixels(param, w * h, use_dma);
341  }
342  else
343  {
344  uint_fast16_t h_step = nogap ? WRITEPIXELS_MAXLEN / w : 1;
345  uint_fast16_t h_len = (h_step > 1) ? ((h - 1) % h_step) + 1 : 1;
346  writePixels(param, w * h_len, use_dma);
347  if (h -= h_len)
348  {
349  param->src_y += h_len;
350  do
351  {
352  param->src_x = src_x;
353  writePixels(param, w * h_step, use_dma);
354  param->src_y += h_step;
355  } while (h -= h_step);
356  }
357  }
358  }
359  else
360  {
361  size_t wb = w * bytes;
362  auto buf = _bus->getDMABuffer(wb);
363  param->fp_copy(buf, 0, w, param);
364  setWindow(x, y, x + w - 1, y + h - 1);
365  write_bytes(buf, wb, true);
366  _has_align_data = (_cfg.dlen_16bit && (_write_bits & 15) && (w & h & 1));
367  while (--h)
368  {
369  param->src_x = src_x;
370  param->src_y++;
371  buf = _bus->getDMABuffer(wb);
372  param->fp_copy(buf, 0, w, param);
373  write_bytes(buf, wb, true);
374  }
375  }
376  }
377  }
378  else
379  {
380  h += y;
381  uint32_t wb = w * bytes;
382  do
383  {
384  uint32_t i = 0;
385  while (w != (i = param->fp_skip(i, w, param)))
386  {
387  auto buf = _bus->getDMABuffer(wb);
388  int32_t len = param->fp_copy(buf, 0, w - i, param);
389  setWindow(x + i, y, x + i + len - 1, y);
390  write_bytes(buf, len * bytes, true);
391  if (w == (i += len)) break;
392  }
393  param->src_x = src_x;
394  param->src_y++;
395  } while (++y != h);
396  }
397  }

466  y += _rowstart + (_rowstart << 16);
