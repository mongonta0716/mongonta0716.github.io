146  void calcClockDiv(uint32_t* div_a, uint32_t* div_b, uint32_t* div_n, uint32_t* clkcnt, uint32_t baseClock, uint32_t targetFreq)
147  {
148  uint32_t diff = INT32_MAX;
149  *div_n = 256;
150  *div_a = 63;
151  *div_b = 62;
152  *clkcnt = 64;
153  uint32_t start_cnt = std::min<uint32_t>(64u, (baseClock / (targetFreq * 2) + 1));
154  uint32_t end_cnt = std::max<uint32_t>(2u, baseClock / 256u / targetFreq);
155  if (start_cnt <= 2) { end_cnt = 1; }
156  for (uint32_t cnt = start_cnt; diff && cnt >= end_cnt; --cnt)
157  {
158  float fdiv = (float)baseClock / cnt / targetFreq;
159  uint32_t n = std::max<uint32_t>(2u, (uint32_t)fdiv);
160  fdiv -= n;
161 
162  for (uint32_t a = 63; diff && a > 0; --a)
163  {
164  uint32_t b = roundf(fdiv * a);
165  if (a == b && n == 256) {
166  break;
167  }
168  uint32_t freq = baseClock / ((n * cnt) + (float)(b * cnt) / (float)a);
169  uint32_t d = abs((int)targetFreq - (int)freq);
170  if (diff <= d) { continue; }
171  diff = d;
172  *clkcnt = cnt;
173  *div_n = n;
174  *div_b = b;
175  *div_a = a;
176  if (b == 0 || a == b) {
177  break;
178  }
179  }
180  }
181  if (*div_a == *div_b)
182  {
183  *div_b = 0;
184  *div_n += 1;
185  }
186  }

870  static cpp::result<void, error_t> i2c_wait(int i2c_port, bool flg_stop = false)
871  {
872  if (i2c_context[i2c_port].state.has_error()) { return cpp::fail(i2c_context[i2c_port].state.error()); }
873  cpp::result<void, error_t> res = {};
874  if (i2c_context[i2c_port].state == i2c_context_t::state_disconnect) { return res; }
875  auto dev = getDev(i2c_port);
876  typeof(dev->int_raw) int_raw;
877  static constexpr uint32_t intmask = I2C_ACK_ERR_INT_RAW_M | I2C_END_DETECT_INT_RAW_M | I2C_ARBITRATION_LOST_INT_RAW_M;
878 
879  if (i2c_context[i2c_port].wait_ack_stage)
880  {
881  int_raw.val = dev->int_raw.val;
882  if (!(int_raw.val & intmask))
883  {
884  uint32_t start_us = lgfx::micros();
885  uint32_t us;
886 #if defined ( CONFIG_IDF_TARGET_ESP32C3 )
887  uint32_t us_limit = (dev->scl_high_period.period + dev->scl_low_period.period + 16 ) * (1 + dev->sr.tx_fifo_cnt);
888 #elif defined ( CONFIG_IDF_TARGET_ESP32S3 )
889  uint32_t us_limit = (dev->scl_high_period.scl_high_period + dev->scl_low_period.scl_low_period + 16 ) * (1 + dev->sr.txfifo_cnt);
890 #else
891  uint32_t us_limit = (dev->scl_high_period.period + dev->scl_low_period.period + 16 ) * (1 + dev->status_reg.tx_fifo_cnt);
892 #endif
893  if (i2c_context[i2c_port].wait_ack_stage == 2) {
894  us_limit += 1024;
895  }
896  do
897  {
898  taskYIELD();
899  us = lgfx::micros() - start_us;
900  int_raw.val = dev->int_raw.val;
901  } while (!(int_raw.val & intmask) && (us <= us_limit));
902  }
903  dev->int_clr.val = int_raw.val;
904 #if !defined (CONFIG_IDF_TARGET) || defined (CONFIG_IDF_TARGET_ESP32)
905  if (!int_raw.end_detect || int_raw.ack_err)
906 #elif defined ( CONFIG_IDF_TARGET_ESP32S3 )
907  if (!int_raw.end_detect_int_raw || int_raw.nack_int_raw)
908 #else
909  if (!int_raw.end_detect || int_raw.nack)
910 #endif
911  {
912  res = cpp::fail(error_t::connection_lost);
913  i2c_context[i2c_port].state = cpp::fail(error_t::connection_lost);
914  }
915  i2c_context[i2c_port].wait_ack_stage = 0;
916  }
917 
918  if (flg_stop || res.has_error())
919  {
920 #if defined ( CONFIG_IDF_TARGET_ESP32S3 )
921  if (i2c_context[i2c_port].state == i2c_context_t::state_read || !int_raw.end_detect_int_raw)
922 #else
923  if (i2c_context[i2c_port].state == i2c_context_t::state_read || !int_raw.end_detect)
924 #endif
925  { // force stop
926  i2c_stop(i2c_port);
927  }
928  else
929  {
930  i2c_set_cmd(dev, 0, i2c_cmd_stop, 0);
931  i2c_set_cmd(dev, 1, i2c_cmd_end, 0);
932  dev->ctr.trans_start = 1;
933  static constexpr uint32_t intmask_ = I2C_ACK_ERR_INT_RAW_M | I2C_TIME_OUT_INT_RAW_M | I2C_END_DETECT_INT_RAW_M | I2C_ARBITRATION_LOST_INT_RAW_M | I2C_TRANS_COMPLETE_INT_RAW_M;
934  uint32_t ms = lgfx::millis();
935  taskYIELD();
936  while (!(dev->int_raw.val & intmask_) && ((millis() - ms) < 14));
937 #if !defined (CONFIG_IDF_TARGET) || defined (CONFIG_IDF_TARGET_ESP32)
938  if (res.has_value() && dev->int_raw.ack_err)
939 #elif defined ( CONFIG_IDF_TARGET_ESP32S3 )
940  if (res.has_value() && dev->int_raw.nack_int_raw)
941 #else
942  if (res.has_value() && dev->int_raw.nack)
943 #endif
944  {
945  res = cpp::fail(error_t::connection_lost);
946  }
947  //ESP_LOGI("LGFX", "I2C stop");
948  }
949  i2c_context[i2c_port].load_reg(dev);
950  if (res)
951  {
952  i2c_context[i2c_port].state = i2c_context_t::state_t::state_disconnect;
953  }
954  }
955  return res;
956  }
