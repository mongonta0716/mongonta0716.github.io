377  bool Panel_IT8951::_set_area( uint32_t x, uint32_t y, uint32_t w, uint32_t h)
378  {
379  uint32_t rx, ry, rw, rh;
380  rx = ((_it8951_rotation+1) & 2) ? _width - w - x : x;
381  ry = ( _it8951_rotation & 2) ? _height - h - y : y;
382  rw = w;
383  rh = h;
384  if (_it8951_rotation & 1)
385  {
386  std::swap(rx, ry);
387  std::swap(rw, rh);
388  }
389 
390  _range_new.left = std::min<uint32_t>(_range_new.left , rx );
391  _range_new.right = std::max<uint32_t>(_range_new.right , rx + rw - 1);
392  _range_new.top = std::min<uint32_t>(_range_new.top , ry );
393  _range_new.bottom = std::max<uint32_t>(_range_new.bottom, ry + rh - 1);
394 
395  if (_epd_mode != epd_mode_t::epd_fastest
396  && _range_old.horizon.intersectsWith(rx, rx + rw - 1)
397  && _range_old.vertical.intersectsWith(ry, ry + rh - 1))
398  {
399  _check_afsr();
400  _range_old.left = INT16_MAX;
401  _range_old.top = INT16_MAX;
402  _range_old.right = 0;
403  _range_old.bottom = 0;
404  }
405 
406  uint16_t params[5];
407  params[0] = IT8951_LDIMG_B_ENDIAN << 8 | IT8951_4BPP << 4 | _it8951_rotation;
408  params[1] = x;
409  params[2] = y;
410  params[3] = w;
411  params[4] = h;
412  return _write_args(IT8951_TCON_LD_IMG_AREA, params, 5);
413  }

415  bool Panel_IT8951::_update_raw_area(epd_update_mode_t mode)
416  {
417  if (_range_new.empty()) return false;
418  uint32_t l = _range_new.left;
419  uint32_t r = _range_new.right;
420 
421  // 更新範囲の幅が小さすぎる場合、IT8951がフリーズすることがある。;
422  // 厳密には、範囲の左右端の座標値の下2ビット捨てた場合に同値になる場合、;
423  // かつ、以前の表示更新がまだ動作中で範囲が重なる場合にフリーズする事例がある。;
424  // この分岐でそれを防止する。;
425  if ((l & ~3) == (r & ~3))
426  {
427  if (( l & 3 ) < (3-(r & 3)))
428  {
429  l = (l & ~3) - 1;
430  }
431  else
432  {
433  r = (r + 4) & ~3;
434  }
435  }
436  uint32_t w = r - l + 1;
437  uint16_t params[7];
438  params[0] = l;
439  params[1] = _range_new.top;
440  params[2] = w;
441  params[3] = _range_new.bottom - _range_new.top + 1;
442  params[4] = mode;
443  params[5] = (uint16_t)_tar_memaddr;
444  params[6] = (uint16_t)(_tar_memaddr >> 16);
445  return _write_args(IT8951_I80_CMD_DPY_BUF_AREA, params, 7);
446  }

536  void Panel_IT8951::writeFillRectPreclipped(uint_fast16_t x, uint_fast16_t y, uint_fast16_t w, uint_fast16_t h, uint32_t rawcolor)
537  {
538  if (_it8951_rotation & 4)
539  {
540  y = height() - y - h;
541  }
542  _set_area(x, y, w, h);
543  bgr888_t color = rawcolor;
544  int32_t sum = color.R8() + (color.G8() << 1) + color.B8();
545  _wait_busy();
546  _bus->writeData(0, 16);
547  bool fast = _epd_mode == epd_mode_t::epd_fast || _epd_mode == epd_mode_t::epd_fastest;
548  uint32_t wid = (((x + w + 3) >> 2) - (x >> 2));
549  do
550  {
551  auto btbl = &Bayer[(y & 3) << 2];
552  ++y;
553  uint32_t value;
554  if (fast)
555  {
556  value = (sum + btbl[2]*16 < 512 ? 0 : 0xF000)
557  | (sum + btbl[3]*16 < 512 ? 0 : 0x0F00)
558  | (sum + btbl[0]*16 < 512 ? 0 : 0x00F0)
559  | (sum + btbl[1]*16 < 512 ? 0 : 0x000F);
560  }
561  else
562  {
563  value = std::min<int32_t>(15, (std::max<int32_t>(0, sum + btbl[2])) >> 6) << 12
564  | std::min<int32_t>(15, (std::max<int32_t>(0, sum + btbl[3])) >> 6) << 8
565  | std::min<int32_t>(15, (std::max<int32_t>(0, sum + btbl[0])) >> 6) << 4
566  | std::min<int32_t>(15, (std::max<int32_t>(0, sum + btbl[1])) >> 6) ;
567  }
568  if (_invert) value = ~value;
569  _bus->writeDataRepeat(value, 16, wid);
570  } while (--h);
571  _write_command(IT8951_TCON_LD_IMG_END);
572  }
