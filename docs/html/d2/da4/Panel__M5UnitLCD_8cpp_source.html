86  bool Panel_M5UnitLCD::_check_repeat(uint32_t cmd, uint_fast8_t limit)
87  {
88  switch (_last_cmd & ~7)
89  {
90  default:
91  break;
92  case CMD_WRITE_RAW:
93  case CMD_WRITE_RLE:
94  if ((_buff_free_count > limit) && (_last_cmd == cmd))
95  {
96  --_buff_free_count;
97  return true;
98  }
99  _bus->endTransaction();
100  cs_control(true);
101  _bus->beginTransaction();
102  cs_control(false);
103  break;
104  }
105 
106  _last_cmd = cmd;
107 
108  if (_buff_free_count > limit)
109  {
110  --_buff_free_count;
111  return false;
112  }
113  limit = std::min<uint_fast8_t>(255, limit * 2);
114 
115  size_t retry = 16;
116  _buff_free_count = 255;
117  while (!_bus->writeCommand(CMD_READ_BUFCOUNT, 8) && --retry);
118  if (retry)
119  {
120  retry = 255;
121  do
122  {
123  if (_bus->readBytes((uint8_t*)&_buff_free_count, 1))
124  {
125  if (_buff_free_count >= limit)
126  {
127  break;
128  }
129  lgfx::delay(2);
130  }
131  else
132  {
133  _bus->endRead();
134  _bus->beginRead();
135  }
136  } while (--retry);
137  }
138  _bus->endTransaction();
139  _bus->beginTransaction();
140 
141  return false;
142  }
