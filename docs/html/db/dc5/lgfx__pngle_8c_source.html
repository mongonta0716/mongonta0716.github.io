173 static void make_pixels(pngle_t *pngle, const uint8_t* buf, uint32_t* rgbbuf, size_t len)
174 {
175  size_t depth = pngle->hdr.depth;
176  uint32_t* argb32 = rgbbuf - 1;
177  uint32_t* last = &argb32[len];
178  if (depth >= 8)
179  {
180  size_t add = depth >> 3;
181  switch (pngle->channels)
182  {
183  case 1: // grayscale or palette
184  do
185  {
186  *++argb32 = buf[0];
187  buf += add;
188  } while (argb32 != last);
189  break;
190 
191  case 2: // grayscale with alpha
192  do
193  {
195  *++argb32 = buf[0]*0x01010100 + buf[add];
196  buf += add * 2;
197  } while (argb32 != last);
198  break;
199 
200  case 3: // color
201  do
202  {
203  *++argb32 = 0xFF
204  + (buf[ 0] << 8)
205  + (buf[add ] << 16)
206  + (buf[add*2] << 24);
207  buf += add * 3;
208  } while (argb32 != last);
209  break;
210 
211  default: //4 : color + alpha
212  do
213  {
214  *++argb32 = (buf[ 0] << 8)
215  + (buf[add ] << 16)
216  + (buf[add*2] << 24)
217  + (buf[add*3] );
218  buf += add * 4;
219  } while (argb32 != last);
220  break;
221  }
222  }
223  else
224  {
225  size_t mask = ((1 << depth) - 1);
226  size_t shift = 8;
227  size_t b = buf[0];
228  do
229  {
230  shift -= depth;
231  *++argb32 = (b >> shift) & mask;
232  if (shift == 0) {
233  shift = 8;
234  b = *++buf;
235  }
236  } while (argb32 != last);
237  }
238 
239  argb32 = rgbbuf - 1;
240  // color type: 0000 0111
241  // ^-- indexed color (palette)
242  // ^--- Color
243  // ^---- Alpha channel
244  switch (pngle->hdr.color_type)
245  {
246  case 0: // grayscale
247  {
248  uint32_t tp = pngle->trans_color;
249  uint32_t magni = pngle->magni;
250  do {
251  uint32_t tmp = *++argb32;
252  *argb32 = ((tmp == tp) ? 0 : 0xFF) + (tmp * magni);
253  } while (argb32 != last);
254  }
255  break;
256 
257  case 2: // true color
258  if (pngle->trans_color != LGFX_PNGLE_NON_TRANS_COLOR)
259  {
260  uint32_t tp = pngle->trans_color;
261  do
262  {
263  if (*++argb32 == tp) { ((uint8_t*)argb32)[0] = 0; }
264  } while (argb32 != last);
265  }
266  break;
267 
268  case 3: // indexed color
269  {
270  size_t n_palettes = pngle->n_palettes;
271  uint32_t* palette = ((uint32_t*)(pngle->palette));
272  do
273  { // lookup palette info
274  size_t pidx = *++argb32;
275  *argb32 = pidx < n_palettes ? palette[pidx] : ~0u;
276  } while (argb32 != last);
277  }
278  break;
279 
280 //case 4: // grayscale with alpha ...nothing to do.
281 //case 6: // truecolor with alpha ...nothing to do.
282  default:
283  break;
284  }
285 }

287 static inline int paeth(int a, int b, int c)
288 {
289  int pa = b - c;
290  int pb = a - c;
291  int pc = abs(pa + pb);
292  pa = abs(pa);
293  pb = abs(pb);
294  if (pb < pa) { pa = pb; a = b; }
295  return (pc < pa) ? c : a;
296 }

298 static void set_interlace_pass(pngle_t *pngle, uint_fast8_t pass)
299 {
300  while (pngle->hdr.width <= pgm_read_byte(&interlace_off_x[pass]) || pngle->hdr.height <= pgm_read_byte(&interlace_off_y[pass]))
301  {
302  ++pass;
303  }
304  pngle->interlace_pass = pass;
305  pngle->drawing_y = pgm_read_byte(&interlace_off_y[pass]);
306  size_t div_x = pgm_read_byte(&interlace_div_x[pass]);
307  size_t scanline_pixels = (pngle->hdr.width - pgm_read_byte(&interlace_off_x[pass]) + div_x - 1) / div_x;
308  pngle->scanline_pixels = scanline_pixels;
309  size_t scanline_stride = (scanline_pixels * pngle->channels * pngle->hdr.depth + 7) >> 3;
310  pngle->scanline_stride = scanline_stride;
311  pngle->scanline_remain_bytes_to_render = scanline_stride;
312 }

314 static int pngle_on_data(pngle_t *pngle, uint8_t *lzbuf, size_t len, size_t outbuf_len)
315 {
316  uint_fast8_t bytes_per_pixel = (pngle->channels * pngle->hdr.depth + 7) >> 3; // 1 if depth <= 8
317  size_t filter_type = pngle->filter_type;
318  size_t remain_bytes = pngle->scanline_remain_bytes_to_render;
319 
320 
321  const uint8_t* p = lzbuf;
322  uint8_t* scanline = &(pngle->scanline_buf[bytes_per_pixel]);
323  while (len)
324  {
325  if (filter_type > 4)
326  {
327  filter_type = *p++; // 0 - 4
328  if (filter_type >= 4)
329  {
330  if (filter_type > 4)
331  {
332  debug_printf("[pngle] Invalid filter type is found; 0x%02x\n", *p);
333  return PNGLE_ERROR("Invalid filter type is found");
334  }
335  // Only when using the paeth filter, shift the previous result to the right by bytes_per_pixel. (To refer to the result of the upper left pixel).
336  memmove(scanline, scanline - bytes_per_pixel, bytes_per_pixel + pngle->scanline_stride);
337  }
338  if (--len == 0) { break; }
339  }
340  size_t cidx = pngle->scanline_stride - remain_bytes;
341 
342  size_t l = (len < remain_bytes) ? len : remain_bytes;
343  remain_bytes -= l;
344  len -= l;
345 
346  const uint8_t* newdata = p - cidx;
347  p += l;
348  size_t last = cidx + l;
349 
350  switch (filter_type) {
351  default: memcpy(&scanline[cidx], &newdata[cidx], l); cidx = last; break;
352  case 1: do { scanline[cidx] = newdata[cidx] + scanline[cidx - bytes_per_pixel]; } while (++cidx != last); break;
353  case 2: do { scanline[cidx] += newdata[cidx]; } while (++cidx != last); break;
354  case 3: do { scanline[cidx] = newdata[cidx] + ((scanline[cidx - bytes_per_pixel] + scanline[cidx]) >> 1); } while (++cidx != last); break;
355  case 4: do { scanline[cidx] = newdata[cidx] + paeth(scanline[cidx - bytes_per_pixel], scanline[cidx + bytes_per_pixel], scanline[cidx]); } while (++cidx != last); break;
356  }
357  if (remain_bytes) { break; }
358 
359  remain_bytes = pngle->scanline_stride; // reset
360  filter_type = ~0;
361 
362  uint32_t draw_x = pgm_read_byte(&interlace_off_x[pngle->interlace_pass]);
363  uint32_t div_x = pgm_read_byte(&interlace_div_x[pngle->interlace_pass]);
364  size_t scanline_pixels = pngle->scanline_pixels;
365  size_t out_pos = 0;
366  size_t out_len = ((((scanline_pixels + 7) & ~7) - 1) % outbuf_len) + 1;
367 
368  do
369  {
370  if (out_len > scanline_pixels - out_pos) { out_len = scanline_pixels - out_pos; }
371  make_pixels(pngle, &scanline[(out_pos * pngle->channels * pngle->hdr.depth) >> 3], pngle->out_buf, out_len);
372  pngle->draw_callback(pngle->user_data, draw_x + out_pos * div_x, pngle->drawing_y, div_x, out_len, (const uint8_t*)pngle->out_buf);
373 
374  out_pos += out_len;
375  out_len = outbuf_len;
376  } while (out_pos < scanline_pixels);
377 
378  pngle->drawing_y += pgm_read_byte(&interlace_div_y[pngle->interlace_pass]);
379  if (pngle->drawing_y >= pngle->hdr.height) {
380  if (pngle->interlace_pass >= 6) { return 0; } // Do nothing further
381 
382  // Interlace: Next pass
383  set_interlace_pass(pngle, pngle->interlace_pass + 1);
384  debug_printf("[pngle] interlace pass changed to: %d\n", pngle->interlace_pass);
385  remain_bytes = pngle->scanline_stride; // reset
386  memset(scanline, 0, remain_bytes);
387  }
388  }
389  pngle->scanline_remain_bytes_to_render = remain_bytes;
390  pngle->filter_type = filter_type;
391 
392  return 0;
393 }

395 int lgfx_pngle_prepare(pngle_t *pngle, lgfx_pngle_read_callback_t read_cb, void* user_data)
396 {
397  if (pngle == NULL || read_cb == NULL) { return PNGLE_STATE_ERROR; }
398  if (pngle->palette ) { free(pngle->palette ); pngle->palette = NULL; }
399  if (pngle->scanline_buf ) { free(pngle->scanline_buf ); pngle->scanline_buf = NULL; }
400 
401  pngle->read_callback = read_cb;
402  pngle->user_data = user_data;
403  pngle->n_palettes = 0;
404  pngle->next_out = pngle->lz_buf;
405  pngle->avail_out = TINFL_LZ_DICT_SIZE;
406  pngle->filter_type = ~0;
407  pngle->trans_color = LGFX_PNGLE_NON_TRANS_COLOR;
408  tinfl_init(&pngle->inflator);
409 
410  if (pngle->read_callback(user_data, pngle->read_buf, 29) != 29
411  || memcmp_P(pngle->read_buf, png_header, sizeof(png_header))) return PNGLE_ERROR("Incorrect PNG signature");
412 
413  memcpy(&(pngle->hdr), &pngle->read_buf[16], sizeof(pngle_ihdr_t));
414  pngle->hdr.width = swap32(pngle->hdr.width );
415  pngle->hdr.height = swap32(pngle->hdr.height);
416 
417  debug_printf("[pngle] width : %d\n", pngle->hdr.width );
418  debug_printf("[pngle] height : %d\n", pngle->hdr.height );
419  debug_printf("[pngle] depth : %d\n", pngle->hdr.depth );
420  debug_printf("[pngle] color_type : %d\n", pngle->hdr.color_type );
421  debug_printf("[pngle] compression: %d\n", pngle->hdr.compression);
422  debug_printf("[pngle] filter : %d\n", pngle->hdr.filter );
423  debug_printf("[pngle] interlace : %d\n", pngle->hdr.interlace );
424 
425  if (pngle->hdr.compression != 0) return PNGLE_ERROR("Unsupported compression type in IHDR");
426  if (pngle->hdr.filter != 0) return PNGLE_ERROR("Unsupported filter type in IHDR");
427 
428  /*
429  Color Allowed Interpretation channels
430  Type Bit Depths
431 
432  0 1,2,4,8,16 Each pixel is a grayscale sample. 1 channels (Brightness)
433 
434  2 8,16 Each pixel is an R,G,B triple. 3 channels (R, G, B)
435 
436  3 1,2,4,8 Each pixel is a palette index; 1 channels (palette info)
437  a PLTE chunk must appear.
438 
439  4 8,16 Each pixel is a grayscale sample, 2 channels (Brightness, Alpha)
440  followed by an alpha sample.
441 
442  6 8,16 Each pixel is an R,G,B triple, 4 channels (R, G, B, Alpha)
443  followed by an alpha sample.
444  */
445  // 111
446  // ^-- indexed color (palette)
447  // ^--- Color
448  // ^---- Alpha channel
449  {
450  static const uint8_t table[] PROGMEM =
451  { 1 << 5 | 0b00000 // type 0 : 1 channels (Brightness)
452  , 0b11111
453  , 3 << 5 | 0b00111 // type 2 : 3 channels (R, G, B)
454  , 1 << 5 | 0b10000 // type 3 : 1 channels (palette info)
455  , 2 << 5 | 0b00111 // type 4 : 2 channels (Brightness, Alpha)
456  , 0b11111
457  , 4 << 5 | 0b00111 // type 6 : 4 channels (R, G, B, Alpha)
458  , 0b11111
459  };
460  uint_fast8_t ch_mask = pgm_read_byte(&table[pngle->hdr.color_type & 7]);
461  pngle->channels = ch_mask >> 5;
462  int depth = pngle->hdr.depth;
463  if (pngle->hdr.color_type > 7 || pngle->channels == 0
464  || depth == 0 || (depth & ch_mask) || (depth != (depth & (-depth)))) return PNGLE_ERROR("Incorrect IHDR info");
465 
466  pngle->magni = 0x01010100
467  * ( (depth == 1) ? 0xFF
468  : (depth == 2) ? 0x55
469  : (depth == 4) ? 0x11
470  : 0x01
471  );
472 
473  uint_fast8_t bytes_per_pixel = (pngle->channels * pngle->hdr.depth + 7) >> 3;
474  size_t memlen = ((pngle->hdr.width * pngle->channels * pngle->hdr.depth + 7) >> 3) + (2 * bytes_per_pixel);
475  if ((pngle->scanline_buf = (uint8_t*)PNGLE_MALLOC(memlen, 1, "scanline flipbuf")) == NULL) return PNGLE_ERROR("Insufficient memory");
476  memset(pngle->scanline_buf, 0, memlen);
477  }
478  // interlace
479  set_interlace_pass(pngle, pngle->hdr.interlace ? 0 : 7);
480 
481  return 0;
482 }

484 int lgfx_pngle_decomp(pngle_t *pngle, lgfx_pngle_draw_callback_t draw_cb)
485 {
486  if (pngle == NULL || draw_cb == NULL) { return PNGLE_STATE_ERROR; }
487  pngle->draw_callback = draw_cb;
488 
489  uint8_t* read_buf = pngle->read_buf;
490  for (;;)
491  {
492  if (pngle->read_callback(pngle->user_data, read_buf, 12) != 12) { return PNGLE_ERROR("Insufficient data"); }
493 
494  uint32_t chunk_remain = read_uint32(&read_buf[4]);
495  uint32_t chunk_type = read_uint32(&read_buf[8]);
496 
497  debug_printf("[pngle] Chunk '%.4s' len %u\n", read_buf + 8, chunk_remain);
498 
499  // initialize & sanity check
500  switch (chunk_type) {
501  default:
502  if (pngle->read_callback(pngle->user_data, NULL, chunk_remain) != chunk_remain) { return PNGLE_ERROR("Insufficient data"); }
503  break;
504 
505  case PNGLE_CHUNK_IDAT:
506  if (chunk_remain <= 0) return PNGLE_ERROR("Invalid IDAT chunk size");
507 
508  do
509  {
510  size_t len = pngle->read_callback(pngle->user_data, read_buf, (chunk_remain < LGFX_PNGLE_READBUF_LEN) ? chunk_remain : LGFX_PNGLE_READBUF_LEN);
511  if (len == 0) { return PNGLE_ERROR("Insufficient data"); }
512  chunk_remain -= len;
513 
514  debug_printf("[pngle] Reading IDAT (len %zd / chunk remain %u)\n", len, chunk_remain);
515 
516  //debug_printf("[pngle] in_bytes %zd, out_bytes %zd, next_out %p\n", in_bytes, out_bytes, pngle->next_out);
517  size_t in_pos = 0;
518  do
519  {
520  size_t in_bytes = len;
521  size_t out_bytes = pngle->avail_out;
522 
523  // XXX: tinfl_decompress always requires (next_out - lz_buf + avail_out) == TINFL_LZ_DICT_SIZE
524  tinfl_status status = tinfl_decompress(&pngle->inflator, (const mz_uint8*)&read_buf[in_pos], &in_bytes, pngle->lz_buf, (mz_uint8*)pngle->next_out, &out_bytes, TINFL_FLAG_HAS_MORE_INPUT | TINFL_FLAG_PARSE_ZLIB_HEADER);
525  if (status < TINFL_STATUS_DONE)
526  {
527  // Decompression failed.
528  debug_printf("[pngle] tinfl_decompress() failed with status %d!\n", status);
529  return PNGLE_ERROR("Failed to decompress the IDAT stream");
530  }
531 
532  len -= in_bytes;
533  in_pos += in_bytes;
534 
535  //debug_printf("[pngle] tinfl_decompress\n");
536  //debug_printf("[pngle] => in_bytes %zd, out_bytes %zd, next_out %p, status %d\n", in_bytes, out_bytes, pngle->next_out, status);
537 
538  if (out_bytes)
539  {
540  if (pngle_on_data(pngle, pngle->next_out, out_bytes, (LGFX_PNGLE_OUTBUF_LEN >> 2) + (len ? in_pos >> 2 : (LGFX_PNGLE_READBUF_LEN >> 2))) < 0) return -1;
541  }
542  pngle->next_out += out_bytes;
543  pngle->avail_out -= out_bytes;
544  // debug_printf("[pngle] => avail_out %zd, next_out %p\n", pngle->avail_out, pngle->next_out);
545  if (pngle->avail_out == 0 || status == TINFL_STATUS_DONE)
546  { // Output buffer is full, or decompression is done, so write buffer to output file.
547  pngle->avail_out = TINFL_LZ_DICT_SIZE;
548  pngle->next_out = pngle->lz_buf;
549  }
