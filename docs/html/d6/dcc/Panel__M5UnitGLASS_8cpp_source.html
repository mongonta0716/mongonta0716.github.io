87  void Panel_M5UnitGlass::display(uint_fast16_t x, uint_fast16_t y, uint_fast16_t w, uint_fast16_t h)
88  {
89  if (0 < w && 0 < h)
90  {
91  _range_mod.left = std::min<int_fast16_t>(_range_mod.left , x );
92  _range_mod.right = std::max<int_fast16_t>(_range_mod.right , x + w - 1);
93  _range_mod.top = std::min<int_fast16_t>(_range_mod.top , y );
94  _range_mod.bottom = std::max<int_fast16_t>(_range_mod.bottom, y + h - 1);
95  }
96  if (_range_mod.empty()) { return; }
97 
98  uint_fast8_t xs = _range_mod.left;
99  uint_fast8_t xe = _range_mod.right;
100  uint_fast8_t ys = _range_mod.top >> 3;
101  uint_fast8_t ye = _range_mod.bottom >> 3;
102 
103  _range_mod.top = INT16_MAX;
104  _range_mod.left = INT16_MAX;
105  _range_mod.right = 0;
106  _range_mod.bottom = 0;
107 
108  y = ys;
109  waitBusy();
110  do
111  {
112  x = xs;
113  uint_fast8_t x_start = x;
114  int index = 0;
115  auto buf = &_buf[y * _cfg.panel_width];
116  do
117  {
118  if (index == 0) {
119  while (0 == (_modified_flags[y] & 0x8000 >> (x >> 3)))
120  {
121  x = 8 + (x & ~7u);
122  if (x > xe) { break; }
123  }
124  x_start = x;
125  }
126  if (x <= xe) {
127  _bus->beginTransaction();
128  _bus->writeCommand(REG_INDEX_PICTURE_BUFFER | index << 8 | buf[x] << 24 , 32);
129  _bus->endTransaction();
130 
131  ++index;
132  ++x;
133 
134  bool flg_draw_picture = (x > xe);
135  if (!flg_draw_picture && 0 == (x & 7)) {
136  flg_draw_picture = (0 == (_modified_flags[y] & 0x8000 >> (x >> 3)));
137  }
138  if (flg_draw_picture) {
139  _bus->beginTransaction();
140  _bus->writeCommand(REG_INDEX_DRAW_PICTURE | x_start << 8 | y << (3 + 16), 24);
141  _bus->writeCommand(index | 1 << (3+8) | 1 << 16, 24);
142  _bus->endTransaction();
143  lgfx::delayMicroseconds(index << 5); // 描画が終わるまで少し待つ
144  index = 0;
145  }
146  }
147  } while (x <= xe);
148  _modified_flags[y] = 0;
149  } while (++y <= ye);
150 
151  _bus->beginTransaction();
152  _bus->writeCommand(REG_INDEX_SHOW | 0x100, 16);
153  _bus->endTransaction();
154  _msec_busy = lgfx::millis() + 5;
155  }
