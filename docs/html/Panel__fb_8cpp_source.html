39  void Panel_fb::fb_draw_rgb_pixel(int x, int y, uint32_t rawcolor)
40  {
41  unsigned int pix_offset = 0;
42  uint8_t r = 0, g = 0, b = 0;
43 
44  // GGGBBBBB RRRRRGGG
45  if (_write_depth = color_depth_t::rgb565_2Byte)
46  {
47  b = (uint8_t)((rawcolor >> 8) & 0b11111);
48  g = (uint8_t)((((rawcolor >> 10) & 0b111000)) | (rawcolor & 0b111));
49  r = (uint8_t)((rawcolor >> 3) & 0b11111);
50 
51  pix_offset = x * 2 + y * _fix_info.line_length;
52  unsigned short c = (r << 11) | (g << 5) | b;
53  // write 'two bytes at once'
54  *((unsigned short*)(_fbp + pix_offset)) = c;
55  }
56  // BBBBBBBB GGGGGGGG RRRRRRRR
57  else if (_write_depth = color_depth_t::rgb888_3Byte)
58  {
59  b = (uint8_t)((rawcolor >> 16) & 0xff);
60  g = (uint8_t)((rawcolor >> 8) & 0xff);
61  r = (uint8_t)((rawcolor >> 0) & 0xff);
62 
63  pix_offset = x * 3 + y * _fix_info.line_length;
64  *((char*)(_fbp + pix_offset)) = b;
65  *((char*)(_fbp + pix_offset + 1)) = g;
66  *((char*)(_fbp + pix_offset + 2)) = r;
67  }
68  }

108  bool Panel_fb::init(bool use_reset)
109  {
110  _fbfd = open(_config_detail.device_name, O_RDWR);
111  if (_fbfd == -1) {
112  // detect target framebuffer.
113  DIR* sysfs_graphics = opendir("/sys/class/graphics");
114  struct dirent* entry;
115  std::string path;
116  std::string target = "/dev/fb0";
117  while((entry = readdir(sysfs_graphics)) != NULL) {
118  if( entry->d_type == DT_LNK ) {
119  path = "/sys/class/graphics/";
120  path.append(entry->d_name);
121  path.append("/name");
122  std::ifstream fs(path.c_str());
123  if( !fs.is_open() ) continue;
124  std::stringstream buffer;
125  buffer << fs.rdbuf();
126  if( buffer.str().find_first_of(_config_detail.device_name) != std::string::npos ) {
127  target = "/dev/";
128  target.append(entry->d_name);
129  break;
130  }
131  }
132  }
133  closedir(sysfs_graphics);
134 
135  _fbfd = open(target.c_str(), O_RDWR);
136  if (_fbfd == -1) {
137  printf("Error: cannot open framebuffer device.\n");
138  return false;
139  }
140  }
141  // printf("The framebuffer device was opened successfully.\n");
142 
143  // Get variable screen information
144  if (ioctl(_fbfd, FBIOGET_VSCREENINFO, &_var_info)) {
145  printf("Error reading variable information.\n");
146  return false;
147  }
148  // printf("%dx%d, %dbpp\n", _var_info.xres, _var_info.yres, _var_info.bits_per_pixel);
149 
150  // 16/24/32
151  setColorDepth((color_depth_t)_var_info.bits_per_pixel);
152 
153  // Get fixed screen information
154  if (ioctl(_fbfd, FBIOGET_FSCREENINFO, &_fix_info)) {
155  printf("Error reading fixed information.\n");
156  return false;
157  }
158 
159  // Figure out the size of the screen in bytes
160  _screensize = _fix_info.smem_len; //finfo.line_length * vinfo.yres;
161 
162  // Map the device to memory
163  _fbp = (char *)mmap(0, _screensize, PROT_READ | PROT_WRITE, MAP_SHARED, _fbfd, 0);
164  if((intptr_t)_fbp == -1) {
165  perror("Error: failed to map framebuffer device to memory");
166  return false;
167  }
168  memset(_fbp, 0, _screensize);
169 
170  return Panel_Device::init(use_reset);
171  }

279  void Panel_fb::writeBlock(uint32_t rawcolor, uint32_t length)
280  {
281  do
282  {
283  uint32_t h = 1;
284  auto w = std::min<uint32_t>(length, _xe + 1 - _xpos);
285  if (length >= (w << 1) && _xpos == _xs)
286  {
287  h = std::min<uint32_t>(length / w, _ye + 1 - _ypos);
288  }
289  writeFillRectPreclipped(_xpos, _ypos, w, h, rawcolor);
290  if ((_xpos += w) <= _xe) return;
291  _xpos = _xs;
292  if (_ye < (_ypos += h)) { _ypos = _ys; }
293  length -= w * h;
294  } while (length);

339  auto k = _cfg.panel_width * bits >> 3;
340 
341  uint_fast8_t r = _internal_rotation;
342  if (!r)
343  {
344  uint_fast16_t linelength;
345  do {
346  linelength = std::min<uint_fast16_t>(xe - x + 1, length);
347  param->fp_copy(&_fbp[y * k], x, x + linelength, param);
348  if ((x += linelength) > xe)
349  {
350  x = xs;
351  y = (y != ye) ? (y + 1) : ys;
352  }
353  } while (length -= linelength);
354  _xpos = x;
355  _ypos = y;
356  return;
357  }
358 
359  int_fast16_t ax = 1;
360  int_fast16_t ay = 1;
361  if ((1u << r) & 0b10010110) { y = _height - (y + 1); ys = _height - (ys + 1); ye = _height - (ye + 1); ay = -1; }
362  if (r & 2) { x = _width - (x + 1); xs = _width - (xs + 1); xe = _width - (xe + 1); ax = -1; }
363  if (param->no_convert)
364  {
365  size_t bytes = bits >> 3;
366  size_t xw = 1;
367  size_t yw = _cfg.panel_width;
368  if (r & 1) std::swap(xw, yw);

370  auto data = (uint8_t*)param->src_data;
371  do
372  {

383  }
384  else

387  y = (y != ye) ? (y + ay) : ys;
388  idx = y * yw + x * xw;
389  }
390  } while (--length);
391  }
392  else
393  {
394  if (r & 1)
395  {
396  do

398  param->fp_copy(&_fbp[x * k], y, y + 1, param);
399  if (x != xe)

401  x += ax;
402  }
403  else
