127  void Panel_Sprite::drawPixelPreclipped(uint_fast16_t x, uint_fast16_t y, uint32_t rawcolor)
128  {
129  uint_fast8_t r = _rotation;
130  if (r)
131  {
132  if ((1u << r) & 0b10010110) { y = _height - (y + 1); }
133  if (r & 2) { x = _width - (x + 1); }
134  if (r & 1) { std::swap(x, y); }
135  }
136  auto bits = _write_bits;
137  uint32_t index = x + y * _bitwidth;
138  if (bits >= 8)
139  {
140  if (bits == 8)
141  {
142  _img.img8()[index] = rawcolor;
143  }
144  else if (bits == 16)
145  {
146  _img.img16()[index] = rawcolor;
147  }
148  else
149  {
150  _img.img24()[index] = rawcolor;
151  }
152  }
153  else
154  {
155  index *= bits;
156  uint8_t* dst = &_img.img8()[index >> 3];
157  uint8_t mask = (uint8_t)(~(0xFF >> bits)) >> (index & 7);
158  *dst = (*dst & ~mask) | (rawcolor & mask);

162  void Panel_Sprite::writeFillRectPreclipped(uint_fast16_t x, uint_fast16_t y, uint_fast16_t w, uint_fast16_t h, uint32_t rawcolor)
163  {
164  uint_fast8_t r = _rotation;
165  if (r)
166  {
167  if ((1u << r) & 0b10010110) { y = _height - (y + h); }
168  if (r & 2) { x = _width - (x + w); }
169  if (r & 1) { std::swap(x, y); std::swap(w, h); }
170  }
171 
172  uint_fast8_t bits = _write_bits;
173  if (bits >= 8)
174  {
175  if (w > 1)
176  {
177  uint_fast8_t bytes = bits >> 3;
178  uint_fast16_t bw = _bitwidth;
179  uint8_t* dst = &_img[(x + y * bw) * bytes];
180  uint8_t* src = dst;
181  uint_fast16_t add_dst = bw * bytes;
182  uint_fast32_t len = w * bytes;
183  uint_fast32_t w32 = w;
184 
185  if (_img.use_memcpy())
186  {
187  if (w32 != bw)
188  {
189  dst += add_dst;
190  }
191  else
192  {
193  w32 *= h;
194  h = 1;
195  }
196  }
197  else
198  {
199  src = (uint8_t*)alloca(len);
200  ++h;
201  }
202  memset_multi(src, rawcolor, bytes, w32);
203  while (--h)
204  {
205  memcpy(dst, src, len);
206  dst += add_dst;
207  }
208  }
209  else
210  {
211  uint_fast16_t bw = _bitwidth;
212  uint32_t index = x + y * bw;
213  if (bits == 8)
214  {
215  auto img = &_img[index];
216  do { *img = rawcolor; img += bw; } while (--h);
217  }
218  else if (bits == 16)
219  {
220  auto img = &_img.img16()[index];
221  do { *img = rawcolor; img += bw; } while (--h);
222  }
223  else if (bits == 32)
224  {
225  auto img = &_img.img32()[index];
226  do { *img = rawcolor; img += bw; } while (--h);
227  }
228  else
229  {
230  auto img = &_img.img24()[index];
231  do { *img = rawcolor; img += bw; } while (--h);
232  }
233  }
234  }
235  else
236  {
237  x *= bits;
238  w *= bits;
239  uint32_t add_dst = _bitwidth * bits >> 3;
240  uint8_t* dst = &_img[y * add_dst + (x >> 3)];
241  uint32_t len = ((x + w) >> 3) - (x >> 3);
242  uint8_t mask = 0xFF >> (x & 7);
243  if (len)
244  {
245  if (mask != 0xFF)
246  {
247  --len;
248  auto d = dst++;
249  uint8_t mc = rawcolor & mask;
250  auto i = h;
251  do { *d = (*d & ~mask) | mc; d += add_dst; } while (--i);
252  }
253  mask = ~(0xFF>>((x + w) & 7));
254  if (len)
255  {
256  auto d = dst;
257  auto i = h;
258  do { memset(d, rawcolor, len); d += add_dst; } while (--i);
259  dst += len;
260  }
261  if (mask == 0) return;
262  }
263  else
264  {
265  mask ^= mask >> w;
266  }
267  rawcolor &= mask;
268  do { *dst = (*dst & ~mask) | rawcolor; dst += add_dst; } while (--h);
269  }
270  }

272  void Panel_Sprite::writeBlock(uint32_t rawcolor, uint32_t length)
273  {
274  do
275  {
276  uint32_t h = 1;
277  auto w = std::min<uint32_t>(length, _xe + 1 - _xpos);
278  if (length >= (w << 1) && _xpos == _xs)
279  {
280  h = std::min<uint32_t>(length / w, _ye + 1 - _ypos);
281  }
282  writeFillRectPreclipped(_xpos, _ypos, w, h, rawcolor);
283  if ((_xpos += w) <= _xe) return;
284  _xpos = _xs;
285  if (_ye < (_ypos += h)) { _ypos = _ys; }
286  length -= w * h;
287  } while (length);

325  void Panel_Sprite::writePixels(pixelcopy_t* param, uint32_t length, bool use_dma)
326  {
327  (void)use_dma;
328  uint_fast16_t xs = _xs;
329  uint_fast16_t xe = _xe;
330  uint_fast16_t ys = _ys;
331  uint_fast16_t ye = _ye;
332  uint_fast16_t x = _xpos;
333  uint_fast16_t y = _ypos;
334  const size_t bits = _write_bits;
335  auto k = _bitwidth * bits >> 3;
336 
337  uint_fast8_t r = _rotation;
338  if (!r)
339  {
340  uint_fast16_t linelength;
341  do {
342  linelength = std::min<uint_fast16_t>(xe - x + 1, length);
343  param->fp_copy(&_img.img8()[y * k], x, x + linelength, param);
344  if ((x += linelength) > xe)
345  {
346  x = xs;
347  y = (y != ye) ? (y + 1) : ys;
348  }
349  } while (length -= linelength);
350  _xpos = x;
351  _ypos = y;
352  return;
353  }
354 
355  int_fast16_t ax = 1;
356  int_fast16_t ay = 1;
357  if ((1u << r) & 0b10010110) { y = _height - (y + 1); ys = _height - (ys + 1); ye = _height - (ye + 1); ay = -1; }
358  if (r & 2) { x = _width - (x + 1); xs = _width - (xs + 1); xe = _width - (xe + 1); ax = -1; }
359  if (param->no_convert)
360  {
361  size_t bytes = bits >> 3;
362  size_t xw = 1;
363  size_t yw = _bitwidth;
364  if (r & 1) std::swap(xw, yw);
365  size_t idx = y * yw + x * xw;
366  auto data = (uint8_t*)param->src_data;
367  do
368  {
369  auto dst = &_img.img8()[idx * bytes];
370  size_t b = 0;
371  do
372  {
373  dst[b] = *data++;
374  } while (++b < bytes);
375  if (x != xe)
376  {
377  idx += xw * ax;
378  x += ax;
379  }
380  else
381  {
382  x = xs;
383  y = (y != ye) ? (y + ay) : ys;
384  idx = y * yw + x * xw;
385  }
386  } while (--length);
387  }
388  else
389  {
390  if (r & 1)
391  {
392  do
393  {
394  param->fp_copy(&_img.img8()[x * k], y, y + 1, param);
395  if (x != xe)
396  {
397  x += ax;
398  }
399  else
400  {
401  x = xs;
402  y = (y != ye) ? (y + ay) : ys;
403  }
404  } while (--length);
405  }
406  else
407  {
408  do
409  {
410  param->fp_copy(&_img.img8()[y * k], x, x + 1, param);
411  if (x != xe)
412  {
413  x += ax;
414  }
415  else
416  {
417  x = xs;
418  y = (y != ye) ? (y + ay) : ys;
419  }
420  } while (--length);
421  }
422  }
423  if ((1u << r) & 0b10010110) { y = _height - (y + 1); }
424  if (r & 2) { x = _width - (x + 1); }
425  _xpos = x;
426  _ypos = y;
427  }

429  void Panel_Sprite::writeImage(uint_fast16_t x, uint_fast16_t y, uint_fast16_t w, uint_fast16_t h, pixelcopy_t* param, bool)
430  {
431  uint_fast8_t r = _rotation;
432  if (r == 0 && param->transp == pixelcopy_t::NON_TRANSP && param->no_convert && _img.use_memcpy())
433  {
434  auto sx = param->src_x;
435  auto bits = param->src_bits;
436  bool flg_memcpy = (0 == (bits & 7));
437  if (!flg_memcpy)
438  {
439  uint_fast8_t mask = (bits == 1) ? 7
440  : (bits == 2) ? 3
441  : 1;
442  flg_memcpy = (sx & mask) == (x & mask) && (w == this->_panel_width || 0 == (w & mask));
443  }
444  if (flg_memcpy)
445  {
446  auto bw = _bitwidth * bits >> 3;
447  auto dst = &_img[bw * y];
448  auto sw = param->src_bitwidth * bits >> 3;
449  auto src = &((uint8_t*)param->src_data)[param->src_y * sw];
450  if (sw == bw && this->_panel_width == w && sx == 0 && x == 0)
451  {
452  memcpy_P(dst, src, bw * h);
453  return;
454  }
455  y = 0;
456  dst += x * bits >> 3;
457  src += sx * bits >> 3;
458  w = w * bits >> 3;
459  do
460  {
461  memcpy_P(&dst[y * bw], &src[y * sw], w);
462  } while (++y != h);
463  return;
464  }
465  }
466 
467  uint32_t nextx = 0;
468  uint32_t nexty = 1 << pixelcopy_t::FP_SCALE;
469  if (r)
470  {
471  _rotate_pixelcopy(x, y, w, h, param, nextx, nexty);
472  }
473  uint32_t sx32 = param->src_x32;
474  uint32_t sy32 = param->src_y32;
475 
476  y *= _bitwidth;
477  do
478  {
479  int32_t pos = x + y;
480  int32_t end = pos + w;
481  while (end != (pos = param->fp_copy(_img, pos, end, param))
482  && end != (pos = param->fp_skip( pos, end, param)));
483  param->src_x32 = (sx32 += nextx);
484  param->src_y32 = (sy32 += nexty);
485  y += _bitwidth;
486  } while (--h);
487  }

544  {

546  if (0 == r && param->no_convert && _write_bits >= 8)

548  h += y;

552  w *= bytes;

554  memcpy(d, &_img[(x + y * bw) * bytes], w);

557  }
558  else
559  {
560  param->src_bitwidth = _bitwidth;
561  param->src_data = _img;

567  uint32_t addy = param->src_y32_add;
568  uint_fast8_t rb = 1 << r;
569  if (rb & 0b10010110) // case 1:2:4:7:
570  {

573  }

575  {
576  addx = -(int32_t)addx;

578  }
579  if ((r+1) & 2)

581  addy = -(int32_t)addy;

608  void Panel_Sprite::copyRect(uint_fast16_t dst_x, uint_fast16_t dst_y, uint_fast16_t w, uint_fast16_t h, uint_fast16_t src_x, uint_fast16_t src_y)
609  {

611  if (r)
612  {
613  if ((1u << r) & 0b10010110) { src_y = _height - (src_y + h); dst_y = _height - (dst_y + h); }
614  if (r & 2) { src_x = _width - (src_x + w); dst_x = _width - (dst_x + w); }
615  if (r & 1) { std::swap(src_x, src_y); std::swap(dst_x, dst_y); std::swap(w, h); }

618  if (_write_bits < 8) {
619  pixelcopy_t param(_img, _write_depth, _write_depth);
620  param.src_bitwidth = _bitwidth;
621  int32_t add_y = (src_y < dst_y) ? -1 : 1;
622  if (src_y != dst_y) {
623  if (src_y < dst_y) {
624  src_y += h - 1;
625  dst_y += h - 1;
626  }
627  param.src_y = src_y;
628  do
629  {
630  param.src_x = src_x;

632  param.fp_copy(_img, idx, idx + w, &param);
633  dst_y += add_y;

636  } else {

638  auto buf = (uint8_t*)alloca(len);
639  param.src_data = buf;

642  memcpy(buf, &_img[src_y * len], len);
643  param.src_x = src_x;
644  auto idx = dst_x + dst_y * _bitwidth;
645  param.fp_copy(_img, idx, idx + w, &param);
646  dst_y += add_y;
647  src_y += add_y;
648  } while (--h);
649  }
650  }
651  else
652  {
653  size_t bytes = _write_bits >> 3;
654  size_t len = w * bytes;
