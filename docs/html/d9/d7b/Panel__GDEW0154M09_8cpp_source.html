331  void Panel_GDEW0154M09::_exec_transfer(uint32_t cmd, const range_rect_t& range, bool invert)
332  {
333  int32_t xs = range.left & ~7;
334  int32_t xe = range.right & ~7;
335 
336  _wait_busy();
337 
338  //_bus->writeCommand(0x91, 8);
339  //_bus->writeCommand(0x90, 8);
340  _bus->writeCommand(0x91 | 0x90 << 8, 16);
341 
342  _bus->writeData(xs | xe << 8, 16);
343  _bus->writeData(getSwap16(range.top) | getSwap16(range.bottom)<<16, 32);
344  _bus->writeData(1, 8);
345 
346  _wait_busy();
347 
348  _bus->writeCommand(cmd, 8);
349  int32_t w = ((xe - xs) >> 3) + 1;
350  int32_t y = range.top;
351  int32_t add = ((_cfg.panel_width + 7) & ~7) >> 3;
352  auto b = &_buf[xs >> 3];
353  if (invert)
354  {
355  b += y * add;
356  do
357  {
358  int32_t i = 0;
359  do
360  {
361  _bus->writeData(~b[i], 8);
362  } while (++i != w);
363  b += add;
364  } while (++y <= range.bottom);
365  }
366  else
367  {
368  do
369  {
370  _bus->writeBytes(&b[add * y], w, true, true);
371  } while (++y <= range.bottom);
