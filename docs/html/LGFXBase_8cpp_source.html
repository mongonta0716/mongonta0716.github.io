511  void LGFXBase::drawLine(int32_t x0, int32_t y0, int32_t x1, int32_t y1)
512  {
513  bool steep = abs(y1 - y0) > abs(x1 - x0);
514 
515  int32_t xstart = _clip_l;
516  int32_t ystart = _clip_t;
517  int32_t xend = _clip_r;
518  int32_t yend = _clip_b;
519 
520  if (steep)
521  {
522  std::swap(xstart, ystart);
523  std::swap(xend, yend);
524  std::swap(x0, y0);
525  std::swap(x1, y1);
526  }
527  if (x0 > x1)
528  {
529  std::swap(x0, x1);
530  std::swap(y0, y1);
531  }
532  if (x0 > xend || x1 < xstart) return;
533  xend = std::min(x1, xend);
534 
535  int32_t dy = abs(y1 - y0);
536  int32_t ystep = (y1 > y0) ? 1 : -1;
537  int32_t dx = x1 - x0;
538  int32_t err = dx >> 1;
539 
540  while (x0 < xstart || y0 < ystart || y0 > yend)
541  {
542  err -= dy;
543  if (err < 0)
544  {
545  err += dx;
546  y0 += ystep;
547  }
548  if (++x0 > xend) return;
549  }
550  int32_t xs = x0;
551  int32_t dlen = 0;
552  if (ystep < 0) std::swap(ystart, yend);
553  yend += ystep;
554 
555  startWrite();
556  if (steep)
557  {
558  do
559  {
560  ++dlen;
561  if ((err -= dy) < 0)
562  {
563  writeFillRectPreclipped(y0, xs, 1, dlen);
564  err += dx;
565  xs = x0 + 1; dlen = 0; y0 += ystep;
566  if (y0 == yend) break;
567  }
568  } while (++x0 <= xend);
569  if (dlen) writeFillRectPreclipped(y0, xs, 1, dlen);
570  }
571  else
572  {
573  do
574  {
575  ++dlen;
576  if ((err -= dy) < 0)
577  {
578  writeFillRectPreclipped(xs, y0, dlen, 1);
579  err += dx;
580  xs = x0 + 1; dlen = 0; y0 += ystep;
581  if (y0 == yend) break;
582  }
583  } while (++x0 <= xend);
584  if (dlen) writeFillRectPreclipped(xs, y0, dlen, 1);
585  }
586  endWrite();
