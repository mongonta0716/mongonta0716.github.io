33  IMU_Base::imu_spec_t BMI270_Class::begin(I2C_Class* i2c)
34  {
35  if (i2c)
36  {
37  _i2c = i2c;
38  }
39 
40  WhoAmI();
41  if (WhoAmI() != 0x24)
42  {
43  return imu_spec_none;
44  }
45 
46  writeRegister8(CMD_REG_ADDR, SOFT_RESET_CMD); // software reset.
47  {
48  int retry = 16;
49  while (0 == readRegister8(PWR_CONF_ADDR) && --retry);
50  }
51 
52  writeRegister8(PWR_CONF_ADDR, 0x00); // Power save disabled.
53  vTaskDelay(1);
54  bool res = _upload_file(bmi270_config_file, 0, sizeof(bmi270_config_file));
55  writeRegister8(INIT_CTRL_ADDR, 0x01);
56 
57  _init = res;
58  if (!res)
59  {
60  return imu_spec_none;
61  }
62 
63  imu_spec_t spec = (imu_spec_t)(imu_spec_accel | imu_spec_gyro);
64  int retry = 16;
65  do { vTaskDelay(1);}
66  while (0 == readRegister8(INTERNAL_STATUS_ADDR) && --retry);
67  res = retry > 0;
68 
69 // 以下は AUXセンサとしてBMM150が接続されている想定の設定
70  auxSetupMode(0x10); // 0x10 = BMM150 I2C Addr
71  auxWriteRegister8(0x4B, 0x83); // software reset + power on
72  auxReadRegister8(0x40); // 0x40 = WhoAmI
73  auto who_am_i = auxReadRegister8(0x40); // 0x40 = WhoAmI
74  if (who_am_i == 0x32)
75  {
76  auxWriteRegister8(0x4C, 0x38); // normal mode / ODR 30Hz
77  spec = (imu_spec_t)(imu_spec_accel | imu_spec_gyro | imu_spec_mag);
78  }
79  writeRegister8(AUX_IF_CONF_ADDR, 0x4F); // FCU_WRITE_EN + Manual BurstLength 8 + BurstLength 8
80  writeRegister8(AUX_RD_ADDR , 0x42); // 0x42 = BMM150 I2C Data X LSB reg
81 // AUX BMM150設定ここまで
82 
83  writeRegister8(INT_MAP_DATA_ADDR, 0xFF);
84  writeRegister8(PWR_CTRL_ADDR, 0x0F); // temp en | ACC en | GYR en | AUX en
85  return spec;
86  }

120  IMU_Base::imu_spec_t BMI270_Class::getImuRawData(imu_raw_data_t* data) const
121  {
122  imu_spec_t res = imu_spec_none;
123  std::uint8_t intstat = readRegister8(INT_STATUS_1_ADDR);
124  if (intstat & 0xE0)
125  {
126  std::int16_t buf[10];
127  if (readRegister(AUX_X_LSB_ADDR, (std::uint8_t*)buf, 20))
128  {
129  if (intstat & 0x80u)
130  {
131  data->accel.x = buf[4];
132  data->accel.y = buf[5];
133  data->accel.z = buf[6];
134  res = (imu_spec_t)(res | imu_spec_accel);
135  }
136  if (intstat & 0x40u)
137  {
138  data->gyro.x = buf[7];
139  data->gyro.y = buf[8];
140  data->gyro.z = buf[9];
141  res = (imu_spec_t)(res | imu_spec_gyro);
142  }
143  if (intstat & 0x20u)
144  {
145  data->mag.x = buf[0] >> 2;
146  data->mag.y = buf[1] >> 2;
147  data->mag.z = buf[2] & 0xFFFE;
148  res = (imu_spec_t)(res | imu_spec_mag);
149  }
150  }
151  }
152  return res;
153  }
