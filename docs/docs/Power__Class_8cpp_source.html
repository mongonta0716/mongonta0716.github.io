320  void Power_Class::setExtOutput(bool enable, ext_port_mask_t port_mask)
321  {
322  switch (M5.getBoard())
323  {
324 #if defined (CONFIG_IDF_TARGET_ESP32S3)
325  case board_t::board_M5StackCoreS3:
326  {
327  _core_s3_output(_core_s3_bus_en, enable);
328  }
329  break;
330 
331 #elif !defined (CONFIG_IDF_TARGET) || defined (CONFIG_IDF_TARGET_ESP32)
332  case board_t::board_M5Paper:
333  if (enable) { m5gfx::gpio_hi(M5Paper_EXT5V_ENABLE_PIN); }
334  else { m5gfx::gpio_lo(M5Paper_EXT5V_ENABLE_PIN); }
335  break;
336 
337  case board_t::board_M5StackCore2:
338  case board_t::board_M5Tough:
339  if (enable && !Axp192.isACIN() && (0.0f < Axp192.getVBUSCurrent()) && (8 >= Axp192.getBatteryLevel()))
340  {
341  // If ACIN is false and VBUS current is detected and the battery is low, power supply from Core to the outside is inhibited.
342  // This is because supplying power externally consumes battery power when there is no power supply from ACIN and power is supplied from VBUS.
343  // ※ If receiving power from M-Bus, PortA, etc., there is no need to setExtPower to true.
344  ESP_LOGW("Power","setExtPower(true) is canceled.");
345  break;
346  }
347  else
348  {
349  Axp192.writeRegister8(0x90, enable ? 0x02 : 0x07); // GPIO0 : enable=LDO / disable=float
350  }
351  NON_BREAK;
352 
353  case board_t::board_M5StickC:
354  case board_t::board_M5StickCPlus:
355  Axp192.setEXTEN(enable);
356  break;
357 
358  case board_t::board_M5Station:
359  for (int i = 0; i < 5; ++i)
360  {
361  if (port_mask & (1 << i)) { Axp192.setGPIO(i, enable); }
362  }
363  if (port_mask & ext_port_mask_t::ext_USB)
364  {
365  if (enable) { m5gfx::gpio_hi(GPIO_NUM_12); } // GPIO12 = M5Station USB power control
366  else { m5gfx::gpio_lo(GPIO_NUM_12); }
367  }
368  if (port_mask & ext_port_mask_t::ext_MAIN)
369  {
370  Axp192.setEXTEN(enable);
371  }
372 #endif
373 
374  default:
375  break;
376  }
377  }
